Elevator System
State Pattern Strategy (Scheduling) Observer
Key Design Decisions
  • Elevator state (Idle/Moving/DoorsOpen) modelled with State pattern — no if/else chains
  • SCAN (elevator) algorithm: serve requests in direction of travel first
  • ElevatorController picks the best elevator for each request
  • Requests are queued per direction; serviced as elevator passes the floor
  • Observer notifies display panels of floor changes
Key Classes
  • Elevator — state machine, sorted request queue
  • ElevatorController — dispatches requests, picks best car
  • IElevatorState — Idle, MovingUp, MovingDown, DoorsOpen
  • Request — floor + direction value object
  • ISchedulingStrategy — SCAN or nearest-car
Interviewer will ask: Q: How does the SCAN algorithm work? → Elevator travels in one direction, services all pending floors in that direction, then reverses. Like a disk head sweep — minimizes travel distance. Q: How do you pick which elevator to assign a request to? → Score each elevator: idle elevators closest to requested floor win. Moving elevators score well if the floor is in their path. Moving-away elevators score worst. Q: How do you handle emergency/overload? → Separate emergency state that bypasses the queue. Overload detected via weight sensor event — doors reopen, request is rejected.
Patterns: State — elevator lifecycle Strategy — SCAN scheduling Observer — floor display
📖 Pattern Primer State Pattern — object changes behaviour based on internal state
Instead of giant if/else chains checking a status field, extract each state into its own class that implements a shared interface. The object delegates behaviour to the current state object.

// ── Enums ─────────────────────────────────────────────
public enum Direction { Up, Down, Idle }
public enum DoorStatus { Open, Closed }

// ── Request (value object) ─────────────────────────────
public record ElevatorRequest(int Floor, Direction Direction);

// ── State Interface ────────────────────────────────────
public interface IElevatorState
{
    void AddRequest(Elevator elevator, int floor);
    void Step(Elevator elevator);       // called each "tick"
    Direction Direction { get; }
}

// ── Idle State ────────────────────────────────────────
public class IdleState : IElevatorState
{
    public Direction Direction => Direction.Idle;

    public void AddRequest(Elevator e, int floor)
    {
        e.UpQueue.Add(floor);
        Console.WriteLine($"[Elevator {e.Id}] Waking up for floor {floor}");
    }

    public void Step(Elevator e)
    {
        if (e.UpQueue.Count == 0 && e.DownQueue.Count == 0) return;

        int next = e.UpQueue.Count > 0 ? e.UpQueue.Min() : e.DownQueue.Max();
        e.TransitionTo(next > e.CurrentFloor
            ? new MovingUpState()
            : new MovingDownState());
    }
}

// ── Moving Up State ───────────────────────────────────
public class MovingUpState : IElevatorState
{
    public Direction Direction => Direction.Up;

    public void AddRequest(Elevator e, int floor)
    {
        if (floor >= e.CurrentFloor) e.UpQueue.Add(floor);
        else                           e.DownQueue.Add(floor);
    }

    public void Step(Elevator e)
    {
        e.CurrentFloor++;
        Console.WriteLine($"[Elevator {e.Id}] ↑ Floor {e.CurrentFloor}");
        e.NotifyObservers();

        if (e.UpQueue.Contains(e.CurrentFloor))
        {
            e.UpQueue.Remove(e.CurrentFloor);
            e.TransitionTo(new DoorsOpenState());
        }
        else if (e.UpQueue.Count == 0)
        {
            e.TransitionTo(e.DownQueue.Count > 0
                ? new MovingDownState()
                : new IdleState());
        }
    }
}

// ── Moving Down State ─────────────────────────────────
public class MovingDownState : IElevatorState
{
    public Direction Direction => Direction.Down;

    public void AddRequest(Elevator e, int floor)
    {
        if (floor <= e.CurrentFloor) e.DownQueue.Add(floor);
        else                           e.UpQueue.Add(floor);
    }

    public void Step(Elevator e)
    {
        e.CurrentFloor--;
        Console.WriteLine($"[Elevator {e.Id}] ↓ Floor {e.CurrentFloor}");
        e.NotifyObservers();

        if (e.DownQueue.Contains(e.CurrentFloor))
        {
            e.DownQueue.Remove(e.CurrentFloor);
            e.TransitionTo(new DoorsOpenState());
        }
        else if (e.DownQueue.Count == 0)
        {
            e.TransitionTo(e.UpQueue.Count > 0
                ? new MovingUpState()
                : new IdleState());
        }
    }
}

// ── Doors Open State ──────────────────────────────────
public class DoorsOpenState : IElevatorState
{
    private int _ticks = 0;
    public Direction Direction => Direction.Idle;

    public void AddRequest(Elevator e, int floor)
    {
        if (floor >= e.CurrentFloor) e.UpQueue.Add(floor);
        else                           e.DownQueue.Add(floor);
    }

    public void Step(Elevator e)
    {
        if (++_ticks < 2) { Console.WriteLine($"[Elevator {e.Id}] 🚪 Doors open at {e.CurrentFloor}"); return; }
        Console.WriteLine($"[Elevator {e.Id}] 🚪 Doors closing at {e.CurrentFloor}");
        e.TransitionTo(e.UpQueue.Count > 0 ? new MovingUpState()
                      : e.DownQueue.Count > 0 ? new MovingDownState()
                      : new IdleState());
    }
}

public class DisplayPanel : IFloorObserver

// ── Elevator ──────────────────────────────────────────
public class Elevator
{
    public int           Id           { get; }
    public int           CurrentFloor { get; set; } = 1;
    public SortedSet<int> UpQueue      { get; } = new();
    public SortedSet<int> DownQueue    { get; } = new(Comparer<int>.Create((a, b) => b - a));

    private IElevatorState         _state     = new IdleState();
    private readonly List<IFloorObserver> _observers = new();

    public Direction Direction => _state.Direction;

    public Elevator(int id) => Id = id;

    public void Subscribe(IFloorObserver obs) => _observers.Add(obs);

    public void NotifyObservers()
    {
        foreach (var obs in _observers)
            obs.OnFloorChanged(Id, CurrentFloor, _state.Direction);
    }

    public void TransitionTo(IElevatorState newState) => _state = newState;

    public void RequestFloor(int floor) => _state.AddRequest(this, floor);

    public void Step() => _state.Step(this);

    public int CostTo(int floor)
    {
        // Lower cost = better candidate for assignment
        int dist = Math.Abs(CurrentFloor - floor);
        if (_state.Direction == Direction.Idle) return dist;

        bool inPath = (_state.Direction == Direction.Up   && floor >= CurrentFloor)
                   || (_state.Direction == Direction.Down && floor <= CurrentFloor);
        return inPath ? dist : dist + 20; // penalty for out-of-path
    }
}

// ── Elevator Controller ───────────────────────────────
public class ElevatorController
{
    private readonly List<Elevator> _elevators = new();

    public void AddElevator(Elevator e) => _elevators.Add(e);

    public void RequestElevator(int floor, Direction dir)
    {
        Console.WriteLine($"🔔 Request: Floor {floor}, Going {dir}");
        var best = _elevators
            .OrderBy(e => e.CostTo(floor))
            .First();
        best.RequestFloor(floor);
    }

    public void Tick()
    {
        foreach (var e in _elevators)
            e.Step();
    }
}

// ── Setup & Usage ─────────────────────────────────────
var ctrl   = new ElevatorController();
var e1     = new Elevator(1);
var e2     = new Elevator(2) { CurrentFloor = 5 };
var panel  = new DisplayPanel();

e1.Subscribe(panel);
e2.Subscribe(panel);
ctrl.AddElevator(e1);
ctrl.AddElevator(e2);

ctrl.RequestElevator(7, Direction.Up);    // e2 wins (closer)
ctrl.RequestElevator(2, Direction.Down);  // e1 wins (idle at 1)

for (int i = 0; i < 10; i++) ctrl.Tick();

Meeting Scheduler
Interval Overlap Repository Pattern Observer (notifications)
Key Design Decisions
  • Interval overlap check: two ranges overlap if start1 < end2 && start2 < end1
  • Each user owns a calendar — meetings stored sorted by start time
  • Finding a free slot: merge all busy intervals, scan for gaps
  • Notification sent to all attendees on booking/cancellation
  • Rooms are a separate resource with their own availability
Key Classes
  • Meeting — title, time range, attendees, room
  • User / Calendar — owns sorted list of meetings
  • Room — capacity, availability check
  • MeetingScheduler — book, cancel, findFreeSlot
  • INotificationService — email/push decoupled via interface
Interviewer will ask: Q: How do you find a free slot for N attendees? → Collect all busy intervals from every attendee. Merge overlapping intervals into one list. Scan for a gap of the required duration between consecutive merged intervals. Q: How do you handle recurring meetings? → Add a RecurrenceRule (daily/weekly/monthly + endDate). On scheduling, generate all occurrences up to endDate and validate each one individually. Q: How do you prevent double-booking a room? → Room keeps its own sorted list of bookings. Check overlap before confirming. Lock the room check + booking in a single transaction (or use a mutex in multi-threaded systems).
Patterns: Repository — CalendarRepository Observer — NotificationService Value Object — TimeSlot

// ── TimeSlot (value object) ───────────────────────────
public record TimeSlot(DateTime Start, DateTime End)
{
    public bool OverlapsWith(TimeSlot other) =>
        Start < other.End && other.Start < End;

    public TimeSpan Duration => End - Start;
}

// ── Room ──────────────────────────────────────────────
public class Room
{
    public Guid         Id       { get; } = Guid.NewGuid();
    public string       Name     { get; }
    public int          Capacity { get; }
    private readonly List<TimeSlot> _bookings = new();

    public Room(string name, int capacity) => (Name, Capacity) = (name, capacity);

    public bool IsAvailable(TimeSlot slot) =>
        !_bookings.Any(b => b.OverlapsWith(slot));

    public void Book(TimeSlot slot)   => _bookings.Add(slot);
    public void Release(TimeSlot slot) => _bookings.Remove(slot);
}

    public string Name  { get; }
    public string Email { get; }
    public Calendar Calendar { get; } = new();

    public User(string name, string email) => (Name, Email) = (name, email);

public class Calendar
{
    private readonly List<Meeting> _meetings = new();

    public bool IsFreeAt(TimeSlot slot) =>
        !_meetings.Any(m => m.Slot.OverlapsWith(slot));

    public void AddMeeting(Meeting m) => _meetings.Add(m);
    public void RemoveMeeting(Guid id) => _meetings.RemoveAll(m => m.Id == id);

    public IReadOnlyList<TimeSlot> GetBusySlots() =>
        _meetings.Select(m => m.Slot).OrderBy(s => s.Start).ToList();

// ── Meeting Scheduler (Facade) ────────────────────────
public class MeetingScheduler
{
    private readonly INotificationService   _notifications;
    private readonly Dictionary<Guid, Meeting> _meetings = new();

    public MeetingScheduler(INotificationService notifications)
        => _notifications = notifications;

    public Meeting? Book(string title, TimeSlot slot, Room room,
                        User organizer, List<User> attendees)
    {
        var allAttendees = attendees.Append(organizer);

        // Check all attendees are free
        if (allAttendees.Any(u => !u.Calendar.IsFreeAt(slot)))
        {
            Console.WriteLine("❌ Conflict: one or more attendees are busy.");
            return null;
        }

        // Check room is free
        if (!room.IsAvailable(slot))
        {
            Console.WriteLine($"❌ Room {room.Name} is already booked.");
            return null;
        }

        var meeting = new Meeting(title, slot, room, organizer, attendees);
        room.Book(slot);

        foreach (var u in allAttendees)
        {
            u.Calendar.AddMeeting(meeting);
            _notifications.Notify(u, $"Meeting '{title}' scheduled at {slot.Start:HH:mm}");
        }

        _meetings[meeting.Id] = meeting;
        Console.WriteLine($"✅ Booked: {title} in {room.Name}");
        return meeting;
    }

    public void Cancel(Guid meetingId)
    {
        if (!_meetings.TryGetValue(meetingId, out var m)) return;

        m.Room.Release(m.Slot);
        foreach (var u in m.Attendees.Append(m.Organizer))
        {
            u.Calendar.RemoveMeeting(meetingId);
            _notifications.Notify(u, $"Meeting '{m.Title}' has been cancelled.");
        }
        _meetings.Remove(meetingId);
        Console.WriteLine($"🗑 Cancelled: {m.Title}");
    }

    public TimeSlot? FindFreeSlot(List<User> users, TimeSpan duration,
                                    DateTime searchFrom, DateTime searchUntil)
    {
        // Merge all busy intervals
        var allBusy = users
            .SelectMany(u => u.Calendar.GetBusySlots())
            .OrderBy(s => s.Start)
            .ToList();

        var merged = new List<TimeSlot>();
        foreach (var slot in allBusy)
        {
            if (merged.Count == 0 || merged[^1].End < slot.Start)
                merged.Add(slot);
            else
                merged[^1] = merged[^1] with { End = DateTime.Max(merged[^1].End, slot.End) };
        }

        // Find first gap ≥ duration between searchFrom and searchUntil
        var cursor = searchFrom;
        foreach (var busy in merged)
        {
            if (cursor + duration <= busy.Start)
                return new TimeSlot(cursor, cursor + duration);
            cursor = DateTime.Max(cursor, busy.End);
        }
        if (cursor + duration <= searchUntil)
            return new TimeSlot(cursor, cursor + duration);

        return null; // no slot found
    }
}

    DateTime.Today.AddHours(9),
    DateTime.Today.AddHours(18));

sched.Cancel(mtg!.Id);

Car Rental System
Strategy (Pricing) Factory (Vehicle) State (Reservation)
Key Design Decisions
  • Vehicle hierarchy: Vehicle → Car, SUV, Truck, Bike
  • Pricing Strategy: base rate × days + extras (insurance, GPS)
  • Reservation states: Pending → Confirmed → Active → Completed / Cancelled
  • Inventory checks vehicle availability for date range before booking
  • Late return fee calculated as additional day multiplied by daily rate
Key Classes
  • Vehicle — abstract; Car/SUV/Truck are concrete
  • Reservation — date range, vehicle, customer, extras
  • IPricingStrategy — daily, weekly-discount, seasonal
  • RentalStore — inventory, book, return, search
  • Extra — GPS, insurance, child-seat (decorator-like)
Interviewer will ask: Q: How do you search for available cars by type and date? → Filter inventory by VehicleType. For each vehicle, check no existing reservation overlaps the requested date range. Return the filtered list. Q: How do you calculate price with optional extras? → Base price from IPricingStrategy. Each Extra has a flat daily cost. Total = base + sum(extra.DailyRate × days). Simple and extensible without touching Vehicle or Reservation. Q: How would you add loyalty points? → Add Customer.LoyaltyPoints. On reservation complete, award points. Strategy decides how many points per dollar. Redeeming points applies a discount in the pricing step.
Patterns: Strategy — pricing Factory — vehicle creation State — reservation lifecycle

// ── Enums ─────────────────────────────────────────────
public enum VehicleType       { Car, SUV, Truck, Bike }
public enum VehicleStatus     { Available, Rented, Maintenance }
public enum ReservationStatus { Pending, Confirmed, Active, Completed, Cancelled }

// ── Vehicle Hierarchy ─────────────────────────────────
public abstract class Vehicle
{
    public Guid          Id            { get; } = Guid.NewGuid();
    public string        PlateNumber   { get; }
    public string        Model         { get; }
    public VehicleType   Type          { get; }
    public VehicleStatus Status        { get; set; } = VehicleStatus.Available;
    public decimal       DailyRate     { get; }

    protected Vehicle(string plate, string model, VehicleType type, decimal dailyRate)
        => (PlateNumber, Model, Type, DailyRate) = (plate, model, type, dailyRate);
}

public class Car   : Vehicle { public Car  (string p, string m) : base(p, m, VehicleType.Car,   40m) {} }
public class SUV   : Vehicle { public SUV  (string p, string m) : base(p, m, VehicleType.SUV,   70m) {} }
public class Truck : Vehicle { public Truck(string p, string m) : base(p, m, VehicleType.Truck, 90m) {} }

// ── Extras (add-on costs) ─────────────────────────────
public record Extra(string Name, decimal DailyRate);

public static class Extras
{
    public static readonly Extra GPS         = new("GPS",         5m);
    public static readonly Extra Insurance    = new("Insurance",   15m);
    public static readonly Extra ChildSeat    = new("Child Seat",  8m);
}

// ── Pricing Strategy ──────────────────────────────────
public interface IPricingStrategy
{
    decimal Calculate(Vehicle v, int days, IEnumerable<Extra> extras);
}

public class StandardPricing : IPricingStrategy
{
    public decimal Calculate(Vehicle v, int days, IEnumerable<Extra> extras)
    {
        decimal extrasTotal = extras.Sum(e => e.DailyRate) * days;
        return v.DailyRate * days + extrasTotal;
    }
}

public class WeeklyDiscountPricing : IPricingStrategy
{
    public decimal Calculate(Vehicle v, int days, IEnumerable<Extra> extras)
    {
        decimal rate  = days >= 7 ? v.DailyRate * 0.85m : v.DailyRate;
        decimal extra = extras.Sum(e => e.DailyRate) * days;
        return rate * days + extra;
    }
}

// ── Reservation ───────────────────────────────────────
public class Reservation
{
    public Guid               Id       { get; } = Guid.NewGuid();
    public Vehicle            Vehicle  { get; }
    public Customer           Customer { get; }
    public DateTime           PickUp   { get; }
    public DateTime           DropOff  { get; }
    public List<Extra>        Extras   { get; }
    public ReservationStatus  Status   { get; set; } = ReservationStatus.Confirmed;
    public decimal            TotalCost{ get; }
    public int                Days     => (int)Math.Ceiling((DropOff - PickUp).TotalDays);

    public Reservation(Vehicle v, Customer c, DateTime pickUp,
                       DateTime dropOff, List<Extra> extras, IPricingStrategy pricing)
    {
        Vehicle  = v;
        Customer = c;
        PickUp   = pickUp;
        DropOff  = dropOff;
        Extras   = extras;
        TotalCost= pricing.Calculate(v, Days, extras);
    }

    public bool OverlapsWith(DateTime start, DateTime end) =>
        PickUp < end && start < DropOff;
}

// ── Rental Store ──────────────────────────────────────
public class RentalStore
{
    private readonly List<Vehicle>     _fleet        = new();
    private readonly List<Reservation>  _reservations = new();
    private readonly IPricingStrategy  _pricing;

    public RentalStore(IPricingStrategy? pricing = null)
        => _pricing = pricing ?? new WeeklyDiscountPricing();

    public void AddVehicle(Vehicle v) => _fleet.Add(v);

    public IEnumerable<Vehicle> Search(VehicleType type, DateTime start, DateTime end) =>
        _fleet.Where(v => v.Type == type && v.Status == VehicleStatus.Available &&
            !_reservations.Any(r => r.Vehicle == v && r.OverlapsWith(start, end)));

    public Reservation? Book(Customer customer, Vehicle vehicle,
                              DateTime start, DateTime end, List<Extra>? extras = null)
    {
        if (_reservations.Any(r => r.Vehicle == vehicle && r.OverlapsWith(start, end)))
        {
            Console.WriteLine("❌ Vehicle not available for those dates.");
            return null;
        }

        var res = new Reservation(vehicle, customer, start, end, extras ?? new(), _pricing);
        _reservations.Add(res);
        Console.WriteLine($"✅ Reserved {vehicle.Model} for {customer.Name} — ${res.TotalCost:F2} for {res.Days} days");
        return res;
    }

    public void Return(Guid reservationId, DateTime actualReturn)
    {
        var res = _reservations.FirstOrDefault(r => r.Id == reservationId);
        if (res == null) return;

        res.Status = ReservationStatus.Completed;
        decimal lateFee = 0m;
        if (actualReturn > res.DropOff)
        {
            int lateDays = (int)Math.Ceiling((actualReturn - res.DropOff).TotalDays);
            lateFee = lateDays * res.Vehicle.DailyRate * 1.5m; // 1.5× for late
            Console.WriteLine($"⚠️  Late by {lateDays} day(s). Late fee: ${lateFee:F2}");
        }
        Console.WriteLine($"🚗 {res.Vehicle.Model} returned. Total charged: ${res.TotalCost + lateFee:F2}");
    }
}

store.AddVehicle(new SUV  ("KA-02-CD-5678", "Honda CR-V"));

Hotel Management System
Strategy (Pricing) State (Room) Observer (Housekeeping)
Key Design Decisions
  • Room status machine: Available → Booked → CheckedIn → Cleaning → Available
  • Housekeeping notified via Observer on checkout
  • Pricing strategy selectable per booking: standard, weekend-rate, seasonal
  • Bill computed at checkout — adds room nights + service charges
  • Room hierarchy: Room → SingleRoom, DoubleRoom, Suite
Key Classes
  • Room — type, status, price, floor
  • Booking — check-in/out dates, guest, room
  • Guest — name, ID, contact
  • Hotel — inventory, search, book, check-in, check-out
  • IHousekeepingObserver — schedules cleaning after checkout
Interviewer will ask: Q: How do you handle concurrent room bookings? → Wrap the search-and-book step in a lock. Or use optimistic locking: add a Version field to Room; if version changed between read and update, retry. In DB systems, use SELECT FOR UPDATE. Q: How do you generate an itemised bill? → Bill = list of line items. Room nights are the base line. Each service request (room service, laundry) adds a line. Total is the sum. Print as a receipt on checkout. Q: How do you support early check-in or late check-out? → Treat as extra service with a flat surcharge. Check room availability for the overlap period before allowing it.
Patterns: Strategy — pricing rules State — room lifecycle Observer — housekeeping

    public void Subscribe(IHousekeepingObserver h) => _housekeeping.Add(h);

    public IEnumerable<Room> SearchAvailable(RoomType type, DateTime ci, DateTime co) =>
        _rooms.Where(r => r.Type == type && r.Status == RoomStatus.Available &&
            !_bookings.Any(b => b.Room == r && b.OverlapsWith(ci, co)));

    public Booking? Book(Guest guest, Room room, DateTime ci, DateTime co)
    {
        if (_bookings.Any(b => b.Room == room && b.OverlapsWith(ci, co)))
        {
            Console.WriteLine("❌ Room not available for those dates.");
            return null;
        }

        var booking = new Booking(room, guest, ci, co);
        room.Status = RoomStatus.Booked;
        _bookings.Add(booking);
        Console.WriteLine($"✅ Room {room.Number} booked for {guest.Name} ({booking.Nights} nights)");
        return booking;
    }

    public void CheckIn(Guid bookingId)
    {
        var b = _bookings.First(x => x.Id == bookingId);
        b.Room.Status = RoomStatus.OccupiedClean;
        Console.WriteLine($"🔑 {b.Guest.Name} checked into Room {b.Room.Number}");
    }

    public void CheckOut(Guid bookingId)
    {
        var b = _bookings.First(x => x.Id == bookingId);

        // Compute bill night by night using pricing strategy
        for (var night = b.CheckIn; night < b.CheckOut; night = night.AddDays(1))
        {
            decimal rate = _pricing.GetNightRate(b.Room, night);
            b.Bill.Add(new BillItem($"Room {b.Room.Number} – {night:ddd dd MMM}", rate));
        }

        b.PrintBill();

        // Notify housekeeping
        foreach (var h in _housekeeping)
            h.OnRoomVacated(b.Room);
    }

    public void AddServiceCharge(Guid bookingId, string service, decimal amount)
    {
        var b = _bookings.First(x => x.Id == bookingId);
        b.Bill.Add(new BillItem(service, amount));
        Console.WriteLine($"💳 Added charge: {service} — ${amount:F2}");
    }
}

{
    private readonly List<LockerStation>      _stations = new();
    private readonly Dictionary<Guid, (Package, Locker, LockerStation)> _assignments = new();
    private readonly ILockerNotifier         _notifier;
    private const int                          RetentionDays = 3;

    public LockerService(ILockerNotifier notifier) => _notifier = notifier;

    public void AddStation(LockerStation s) => _stations.Add(s);

    public bool DeliverPackage(Package pkg, Guid stationId)
    {
        var station = _stations.FirstOrDefault(s => s.Id == stationId);
        if (station == null) { Console.WriteLine("❌ Station not found."); return false; }

        var locker = station.FindBestLocker(pkg.Size);
        if (locker == null) { Console.WriteLine("❌ No suitable locker available."); return false; }

        pkg.GenerateOtp();
        pkg.Status      = PackageStatus.Delivered;
        pkg.DeliveredAt = DateTime.Now;
        pkg.ExpiresAt   = DateTime.Now.AddDays(RetentionDays);

        locker.Assign(pkg);
        _assignments[pkg.Id] = (pkg, locker, station);

        Console.WriteLine($"📦 Package {pkg.TrackingId} placed in {locker.Size} locker at {station.Address}");
        _notifier.NotifyDelivery(pkg, station, pkg.Otp);
        return true;
    }

    public bool PickupPackage(string trackingId, string otp)
    {
        var match = _assignments.Values
            .FirstOrDefault(a => a.Item1.TrackingId == trackingId);

        if (match == default)
        {
            Console.WriteLine("❌ Package not found.");
            return false;
        }

        var (pkg, locker, _) = match;

        if (pkg.ExpiresAt < DateTime.Now)
        {
            Console.WriteLine("❌ Package has expired.");
            return false;
        }

        if (!pkg.ValidateOtp(otp))
        {
            Console.WriteLine("❌ Incorrect OTP.");
            return false;
        }

        pkg.Status = PackageStatus.PickedUp;
        locker.Release();
        _assignments.Remove(pkg.Id);

        Console.WriteLine($"✅ Package {trackingId} picked up successfully.");
        return true;
    }

    public void ProcessExpiredPackages()
    {
        var now     = DateTime.Now;
        var expired = _assignments.Values
            .Where(a => a.Item1.ExpiresAt < now)
            .ToList();

        foreach (var (pkg, locker, _) in expired)
        {
            pkg.Status = PackageStatus.Expired;
            locker.Release();
            _assignments.Remove(pkg.Id);
            _notifier.NotifyExpiry(pkg);
            Console.WriteLine($"🔄 Expired package {pkg.TrackingId} removed and queued for return.");
        }
    }
}

Interviewer will ask: Q: How does debt simplification work? → Build a net-balance map. Sort into creditors (positive) and debtors (negative). Two-pointer: match biggest creditor with biggest debtor. 📖 Concept Primer Debt Simplification — greedy two-pointer to minimize transactions
Given a set of net balances (positive = owed to you, negative = you owe), find the minimum number of transactions to settle all debts. Greedy: always match the largest creditor with the largest debtor.
// Example: Alice +300, Bob -100, Charlie -200
// Sorted creditors: [Alice +300]
// Sorted debtors:   [Charlie -200, Bob -100]
//
// Step 1: Charlie pays Alice $200 → Alice +100, Charlie 0
// Step 2: Bob pays Alice $100    → Alice 0,    Bob 0
// Result: 2 transactions instead of potentially many more

static List<(string From, string To, decimal Amount)>
    Simplify(Dictionary<string, decimal> balances)
{
    var result  = new List<(string, string, decimal)>();
    var pos     = new SortedList<decimal, string>(); // creditors
    var neg     = new SortedList<decimal, string>(); // debtors

    foreach (var (name, bal) in balances)
    {
        if (bal > 0) pos[bal]  = name;
        if (bal < 0) neg[-bal] = name;
    }
    // Two-pointer: match largest creditor with largest debtor
    // Transfer min(credit, debt), advance whichever hits 0
    return result;
}

// ── Split Types ───────────────────────────────────────
public interface ISplitStrategy
{
    // Returns mapping: userId → how much they owe
    Dictionary<Guid, decimal> Split(decimal totalAmount, List<User> participants, object? metadata = null);
}

public class EqualSplit : ISplitStrategy
{
    public Dictionary<Guid, decimal> Split(decimal total, List<User> participants, object? _)
    {
        decimal share = Math.Round(total / participants.Count, 2);
        return participants.ToDictionary(u => u.Id, _ => share);
    }
}

public class ExactSplit : ISplitStrategy
{
    public Dictionary<Guid, decimal> Split(decimal total, List<User> participants, object? metadata)
    {
        var exactAmounts = (Dictionary<Guid, decimal>)metadata!;
        if (Math.Abs(exactAmounts.Values.Sum() - total) > 0.01m)
            throw new InvalidOperationException("Exact splits must sum to total.");
        return exactAmounts;
    }
}

public class PercentageSplit : ISplitStrategy
{
    public Dictionary<Guid, decimal> Split(decimal total, List<User> participants, object? metadata)
    {
        var percentages = (Dictionary<Guid, decimal>)metadata!;
        if (Math.Abs(percentages.Values.Sum() - 100m) > 0.01m)
            throw new InvalidOperationException("Percentages must sum to 100.");
        return percentages.ToDictionary(
            kv => kv.Key,
            kv => Math.Round(total * kv.Value / 100m, 2));
    }
}

    public User(string name) => Name = name;
}

// ── Expense ───────────────────────────────────────────
public class Expense
{
    public Guid                    Id          { get; } = Guid.NewGuid();
    public string                  Description { get; }
    public decimal                 Amount      { get; }
    public User                    PaidBy      { get; }
    public List<User>              Participants{ get; }
    public ISplitStrategy          Strategy    { get; }
    public object?                 Metadata    { get; }
    public DateTime                CreatedAt   { get; } = DateTime.Now;

    // Computed: who owes the payer what amount
    public Dictionary<Guid, decimal> Splits { get; }

    public Expense(string desc, decimal amount, User paidBy,
                   List<User> participants, ISplitStrategy strategy, object? metadata = null)
    {
        Description  = desc;
        Amount       = amount;
        PaidBy       = paidBy;
        Participants = participants;
        Strategy     = strategy;
        Metadata     = metadata;
        Splits       = strategy.Split(amount, participants, metadata);
    }
}

// ── Group ─────────────────────────────────────────────
public class Group
{
    public Guid         Id       { get; } = Guid.NewGuid();
    public string       Name     { get; }
    public List<User>   Members  { get; } = new();
    public List<Expense> Expenses { get; } = new();

    public Group(string name) => Name = name;

    public void AddMember(User u)   => Members.Add(u);
    public void AddExpense(Expense e) => Expenses.Add(e);

    // Net balance per user within the group (positive = owed to you)
    public Dictionary<Guid, decimal> GetBalances()
    {
        var balances = Members.ToDictionary(u => u.Id, _ => 0m);

        foreach (var exp in Expenses)
        {
            foreach (var (userId, owed) in exp.Splits)
            {
                if (userId == exp.PaidBy.Id) continue; // payer doesn't owe themselves
                balances[exp.PaidBy.Id] += owed;  // payer is owed more
                balances[userId]          -= owed;  // participant owes more
            }
        }
        return balances;
    }
}

// ── Splitwise Service ─────────────────────────────────
public class SplitwiseService
{
    private readonly List<Group> _groups = new();

    public void AddGroup(Group g) => _groups.Add(g);

    public Expense AddExpense(Group group, string desc, decimal amount,
                              User paidBy, List<User> participants,
                              ISplitStrategy strategy, object? metadata = null)
    {
        var expense = new Expense(desc, amount, paidBy, participants, strategy, metadata);
        group.AddExpense(expense);

        Console.WriteLine($"\n💰 Expense: {desc} — ${amount:F2} paid by {paidBy.Name}");
        foreach (var p in participants)
        {
            if (expense.Splits.TryGetValue(p.Id, out var share))
                Console.WriteLine($"  {p.Name} owes ${share:F2}");
        }
        return expense;
    }

    public void ShowBalances(Group group, Dictionary<Guid, User> userMap)
    {
        Console.WriteLine($"\n📊 Balances in '{group.Name}':");
        var balances = group.GetBalances();
        foreach (var (uid, bal) in balances)
        {
            if (!userMap.TryGetValue(uid, out var u)) continue;
            string label = bal > 0 ? $"gets back ${bal:F2}"
                         : bal < 0 ? $"owes ${-bal:F2}"
                         : "is settled up";
            Console.WriteLine($"  {u.Name,-12} {label}");
        }
    }

    /// Greedy debt simplification — minimum number of transactions
    public List<(User From, User To, decimal Amount)> SimplifyDebts(
        Group group, Dictionary<Guid, User> userMap)
    {
        var balances  = group.GetBalances();
        var creditors = new SortedList<decimal, Guid>();
        var debtors   = new SortedList<decimal, Guid>();

        foreach (var (uid, bal) in balances)
        {
            if (bal >  0.01m) creditors[bal]  = uid;
            if (bal < -0.01m) debtors[-bal]   = uid;
        }

        var result   = new List<(User, User, decimal)>();
        var cList    = creditors.ToList();
        var dList    = debtors.ToList();
        int ci = cList.Count - 1, di = dList.Count - 1;

        while (ci >= 0 && di >= 0)
        {
            decimal credit = cList[ci].Key;
            decimal debt   = dList[di].Key;
            decimal settle = Math.Min(credit, debt);

            result.Add((userMap[dList[di].Value], userMap[cList[ci].Value], settle));

            cList[ci] = new(credit - settle, cList[ci].Value);
            dList[di] = new(debt   - settle, dList[di].Value);

            if (cList[ci].Key < 0.01m) ci--;
            if (dList[di].Key < 0.01m) di--;
        }

        Console.WriteLine($"\n✨ Simplified transactions for '{group.Name}':");
        foreach (var (from, to, amt) in result)
            Console.WriteLine($"  {from.Name} → {to.Name}: ${amt:F2}");

        return result;
    }
}

// ── Setup & Usage ─────────────────────────────────────
var alice = new User("Alice");
var bob   = new User("Bob");
var carol = new User("Carol");

var userMap = new Dictionary<Guid, User> {
    [alice.Id] = alice, [bob.Id] = bob, [carol.Id] = carol
};

var group = new Group("Goa Trip");
group.AddMember(alice); group.AddMember(bob); group.AddMember(carol);

var service = new SplitwiseService();
service.AddGroup(group);

// Alice pays ₹900 for hotel — split equally
service.AddExpense(group, "Hotel", 900m, alice,
    new List<User> { alice, bob, carol },
    new EqualSplit());

// Bob pays ₹300 for dinner — exact split
service.AddExpense(group, "Dinner", 300m, bob,
    new List<User> { alice, bob, carol },
    new ExactSplit(),
    new Dictionary<Guid, decimal> {
        [alice.Id] = 150m, [bob.Id] = 100m, [carol.Id] = 50m
    });

// Carol pays ₹600 for activities — percentage split
service.AddExpense(group, "Activities", 600m, carol,
    new List<User> { alice, bob, carol },
    new PercentageSplit(),
    new Dictionary<Guid, decimal> {
        [alice.Id] = 50m, [bob.Id] = 30m, [carol.Id] = 20m
    });

service.ShowBalances(group, userMap);
service.SimplifyDebts(group, userMap);
