Chess Game
Factory — Pieces Command — Move/Undo Memento — History Strategy — Move Rules
Key Design Decisions
  • Each Piece knows its own move rules via GetValidMoves() — Strategy per piece type
  • Move is a Command object — encapsulates Execute + Undo
  • Board is an 8×8 grid of nullable Pieces — simple and traversable
  • Move history is a Stack<Move> — undo is just Pop + reverse
  • PieceFactory builds the initial board layout — one place to change setup
Key Classes
  • Piece — abstract base: Color, Position, GetValidMoves()
  • King / Queen / Rook / Bishop / Knight / Pawn — concrete pieces
  • Board — 8×8 grid, move execution, check detection
  • Move — from/to squares, captured piece (for undo)
  • Game — turn management, check/checkmate detection
  • Player — color + name
Interviewer will ask: Q: How do you implement undo? → Every Move stores the captured piece. Undo moves the piece back to its origin square and restores the captured piece. Stack<Move> history — Pop gives you the last move instantly. Q: How do you detect check? → After every move, collect all valid moves for every opponent piece. If any of them land on the King's square, you're in check. Checkmate = in check AND no valid moves exist that escape it. Q: Where do special rules go (castling, en passant)? → In the relevant piece's GetValidMoves(). King checks castling conditions. Pawn checks en passant on the last move stored in history. Board stays clean.
Patterns: Factory — piece creation Command — Move with Undo Strategy — move rules per piece
📖 Pattern Primer Command Pattern — encapsulate actions as objects, enabling undo
Wrap an action (and its reverse) inside an object. Store those objects in a history stack. Undo = pop and call reverse. Great for editors, games, transaction systems.
public interface ICommand
{
    void Execute();
    void Undo();
}

// A simple text editor example
public class InsertTextCommand : ICommand
{
    private readonly StringBuilder _doc;
    private readonly string _text;
    private readonly int    _pos;

    public InsertTextCommand(StringBuilder doc, string text, int pos)
        => (_doc, _text, _pos) = (doc, text, pos);

    public void Execute() => _doc.Insert(_pos, _text);
    public void Undo()    => _doc.Remove(_pos, _text.Length);
}

public class Editor
{
    private readonly StringBuilder  _doc     = new();
    private readonly Stack<ICommand> _history = new();

    public void Execute(ICommand cmd)
    {
        cmd.Execute();
        _history.Push(cmd);
    }

    public void Undo()
    {
        if (_history.Count > 0)
            _history.Pop().Undo();
    }

    public override string ToString() => _doc.ToString();
}

var ed = new Editor();
ed.Execute(new InsertTextCommand(ed._doc, "Hello", 0));  // "Hello"
ed.Execute(new InsertTextCommand(ed._doc, " World", 5)); // "Hello World"
ed.Undo();                                                // "Hello"

public enum PieceColor { White, Black }
public enum GameStatus { Active, Check, Checkmate, Stalemate, Draw }

public record Position(int Row, int Col)
{
    public bool IsValid => Row >= 0 && Row < 8 && Col >= 0 && Col < 8;
    public override string ToString() =>
        $"{(char)('a' + Col)}{Row + 1}";  // e.g. e4
}

// ── Move (Command Object) ─────────────────────────────
public class Move
{
    public Piece    Piece          { get; }
    public Position From           { get; }
    public Position To             { get; }
    public Piece?   CapturedPiece  { get; }  // needed for undo

    public Move(Piece piece, Position from, Position to, Piece? captured = null)
        => (Piece, From, To, CapturedPiece) = (piece, from, to, captured);

// ── Piece (abstract) ──────────────────────────────────
public abstract class Piece
{
    public PieceColor Color    { get; }
    public Position   Position { get; set; }
    public bool       HasMoved { get; set; }

    protected Piece(PieceColor color, Position position)
        => (Color, Position) = (color, position);

    // Each subclass implements its own movement rules
    public abstract IEnumerable<Position> GetValidMoves(Board board);

    public abstract string Symbol { get; }
}

// ── Concrete Pieces ───────────────────────────────────
public class Rook : Piece
{
    public Rook(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "R" : "r";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        // Rook moves along ranks and files — slide until blocked
        var moves = new List<Position>();
        int[][] dirs = { new[]{1,0}, new[]{-1,0}, new[]{0,1}, new[]{0,-1} };
        foreach (var d in dirs)
            moves.AddRange(board.SlideMoves(Position, d[0], d[1], Color));
        return moves;
    }
}

public class Bishop : Piece
{
    public Bishop(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "B" : "b";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        var moves = new List<Position>();
        int[][] dirs = { new[]{1,1}, new[]{1,-1}, new[]{-1,1}, new[]{-1,-1} };
        foreach (var d in dirs)
            moves.AddRange(board.SlideMoves(Position, d[0], d[1], Color));
        return moves;
    }
}

public class Knight : Piece
{
    public Knight(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "N" : "n";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        // Knight's L-shape offsets — all 8 possible jumps
        int[][] jumps = {
            new[]{2,1}, new[]{2,-1}, new[]{-2,1},  new[]{-2,-1},
            new[]{1,2}, new[]{1,-2}, new[]{-1, 2}, new[]{-1,-2}
        };
        return jumps
            .Select(j => new Position(Position.Row + j[0], Position.Col + j[1]))
            .Where(p => p.IsValid && board.GetPiece(p)?.Color != Color);
    }
}

public class Queen : Piece
{
    public Queen(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "Q" : "q";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        // Queen = Rook + Bishop combined
        var moves = new List<Position>();
        int[][] dirs = {
            new[]{1,0}, new[]{-1,0}, new[]{0,1},  new[]{0,-1},
            new[]{1,1}, new[]{1,-1}, new[]{-1,1}, new[]{-1,-1}
        };
        foreach (var d in dirs)
            moves.AddRange(board.SlideMoves(Position, d[0], d[1], Color));
        return moves;
    }
}

public class King : Piece
{
    public King(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "K" : "k";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        // One step in any direction
        int[][] steps = {
            new[]{1,0}, new[]{-1,0}, new[]{0,1},  new[]{0,-1},
            new[]{1,1}, new[]{1,-1}, new[]{-1,1}, new[]{-1,-1}
        };
        return steps
            .Select(s => new Position(Position.Row + s[0], Position.Col + s[1]))
            .Where(p => p.IsValid && board.GetPiece(p)?.Color != Color);
    }
}

public class Pawn : Piece
{
    public Pawn(PieceColor c, Position p) : base(c, p) { }
    public override string Symbol => Color == PieceColor.White ? "P" : "p";

    public override IEnumerable<Position> GetValidMoves(Board board)
    {
        var moves = new List<Position>();
        int dir = Color == PieceColor.White ? 1 : -1;
        var forward = new Position(Position.Row + dir, Position.Col);

        // Forward move — only if square is empty
        if (forward.IsValid && board.GetPiece(forward) == null)
        {
            moves.Add(forward);
            // Double move from starting rank
            var startRow = Color == PieceColor.White ? 1 : 6;
            var doubleForward = new Position(Position.Row + 2*dir, Position.Col);
            if (!HasMoved && Position.Row == startRow
                && doubleForward.IsValid && board.GetPiece(doubleForward) == null)
                moves.Add(doubleForward);
        }

        // Diagonal captures
        foreach (var dc in new[] { -1, 1 })
        {
            var diag = new Position(Position.Row + dir, Position.Col + dc);
            if (diag.IsValid && board.GetPiece(diag)?.Color == Opponent())
                moves.Add(diag);
        }
        return moves;
    }

    private PieceColor Opponent() =>
        Color == PieceColor.White ? PieceColor.Black : PieceColor.White;
}

// ── Board ─────────────────────────────────────────────
public class Board
{
    private readonly Piece?[,] _grid = new Piece[8, 8];
    public Stack<Move> History { get; } = new();

    public Piece? GetPiece(Position p) => _grid[p.Row, p.Col];

    public void SetPiece(Position p, Piece? piece)
    {
        _grid[p.Row, p.Col] = piece;
        if (piece != null) piece.Position = p;
    }

    // Execute a move and record it in history
    public void ExecuteMove(Move move)
    {
        SetPiece(move.To, move.Piece);
        SetPiece(move.From, null);
        move.Piece.HasMoved = true;
        History.Push(move);
    }

    // Undo the last move — restore piece and any capture
    public void UndoMove()
    {
        if (History.Count == 0) return;
        var move = History.Pop();
        SetPiece(move.From, move.Piece);
        SetPiece(move.To, move.CapturedPiece);
    }

    // Slide in a direction until blocked (Rook, Bishop, Queen)
    public IEnumerable<Position> SlideMoves(Position from, int dr, int dc, PieceColor color)
    {
        var moves = new List<Position>();
        var cur = new Position(from.Row + dr, from.Col + dc);
        while (cur.IsValid)
        {
            var occupant = GetPiece(cur);
            if (occupant == null) { moves.Add(cur); }
            else {
                if (occupant.Color != color) moves.Add(cur); // capture
                break;  // blocked regardless
            }
            cur = new Position(cur.Row + dr, cur.Col + dc);
        }
        return moves;
    }

    // Is a given color's king in check?
    public bool IsInCheck(PieceColor color)
    {
        var king = AllPieces().FirstOrDefault(p => p is King && p.Color == color);
        if (king == null) return false;

        var opponent = color == PieceColor.White ? PieceColor.Black : PieceColor.White;
        return AllPieces()
            .Where(p => p.Color == opponent)
            .SelectMany(p => p.GetValidMoves(this))
            .Any(pos => pos == king.Position);
    }

    public IEnumerable<Piece> AllPieces() =>
        Enumerable.Range(0, 8)
            .SelectMany(r => Enumerable.Range(0, 8).Select(c => _grid[r, c]))
            .Where(p => p != null)!;

    public void Print()
    {
        Console.WriteLine("  a b c d e f g h");
        for (int r = 7; r >= 0; r--)
        {
            Console.Write($"{r + 1} ");
            for (int c = 0; c < 8; c++)
                Console.Write(_grid[r, c]?.Symbol ?? "." + " ");
            Console.WriteLine();
        }
    }
}

// ── Game ──────────────────────────────────────────────
public class Game
{
    public Board     Board   { get; } = new();
    public PieceColor Current { get; private set; } = PieceColor.White;
    public GameStatus Status  { get; private set; } = GameStatus.Active;

    public Game() => SetupBoard();

    public bool TryMove(Position from, Position to)
    {
        var piece = Board.GetPiece(from);
        if (piece == null || piece.Color != Current) return false;
        if (!piece.GetValidMoves(Board).Contains(to))  return false;

        var captured = Board.GetPiece(to);
        var move     = new Move(piece, from, to, captured);
        Board.ExecuteMove(move);

        // Illegal if it leaves own king in check — must undo
        if (Board.IsInCheck(Current))
        {
            Board.UndoMove();
            Console.WriteLine("Illegal — leaves King in check!");
            return false;
        }

        var opponent = Current == PieceColor.White ? PieceColor.Black : PieceColor.White;
        Current  = opponent;
        Status   = ComputeStatus(opponent);
        return true;
    }

    public void Undo() => Board.UndoMove();

    private GameStatus ComputeStatus(PieceColor color)
    {
        bool inCheck  = Board.IsInCheck(color);
        bool hasMoves = Board.AllPieces()
            .Where(p => p.Color == color)
            .SelectMany(p => p.GetValidMoves(Board))
            .Any();

        if (inCheck  && !hasMoves) return GameStatus.Checkmate;
        if (!inCheck && !hasMoves) return GameStatus.Stalemate;
        if (inCheck)               return GameStatus.Check;
        return GameStatus.Active;
    }

    private void SetupBoard()
    {
        // Place pawns
        for (int c = 0; c < 8; c++)
        {
            Board.SetPiece(new Position(1, c), new Pawn(PieceColor.White, new Position(1, c)));
            Board.SetPiece(new Position(6, c), new Pawn(PieceColor.Black, new Position(6, c)));
        }
        // Place back ranks (simplified — just a few key pieces)
        Board.SetPiece(new Position(0, 4), new King(PieceColor.White, new Position(0, 4)));
        Board.SetPiece(new Position(7, 4), new King(PieceColor.Black, new Position(7, 4)));
        Board.SetPiece(new Position(0, 3), new Queen(PieceColor.White, new Position(0, 3)));
        Board.SetPiece(new Position(7, 3), new Queen(PieceColor.Black, new Position(7, 3)));
    }
}

// ── Usage ─────────────────────────────────────────────
var game = new Game();
game.Board.Print();
game.TryMove(new Position(1, 4), new Position(3, 4)); // e2 → e4
Console.WriteLine($"Status: {game.Status}");

ATM
State Pattern ★ Chain of Responsibility Singleton
Key Design Decisions
  • State Pattern drives the ATM: Idle → CardInserted → PinVerified → Transaction
  • Cash dispensing uses Chain of Responsibility (£50 → £20 → £10 → £5 notes)
  • Account and Card are separate — a card references an account
  • Bank is the Singleton system; ATM is a hardware endpoint
  • Balance check happens BEFORE dispensing, inside a single transaction
ATM States
  • IdleState — waiting for card insertion
  • CardInsertedState — card in, awaiting PIN
  • AuthenticatedState — PIN verified, show menu
  • TransactionState — performing withdraw/deposit
Interviewer will ask: Q: How do you prevent two ATMs withdrawing from the same account simultaneously? → Use optimistic concurrency on the Account. Store a RowVersion/ETag. The update query only succeeds if the version matches. If it fails, return InsufficientFunds. At the class level: lock on the account object, or use Interlocked.CompareExchange on the balance. Q: What happens if the machine runs out of £20 notes? → The Chain of Responsibility dispatcher skips that denomination and tries the next. If total can't be made with available notes, the transaction is declined.
Patterns: State ★ — ATM workflow Chain of Responsibility — cash dispenser Singleton — Bank
📖 Pattern Primer Chain of Responsibility — denominations example
Each handler tries to fulfil as much as it can of a request, then passes the remainder down. Perfect for cash dispensing, approval hierarchies, and middleware pipelines.
// Dispensing cash: try £50s first, then £20s, £10s, £5s
public abstract class NoteDispenser
{
    protected NoteDispenser? _next;
    protected int _denomination;
    protected int _count;  // notes in this cassette

    public NoteDispenser SetNext(NoteDispenser next)
        => (_next = next, next).Item2;

    public int Dispense(int amount)
    {
        int notesNeeded = amount / _denomination;
        int notesUsed   = Math.Min(notesNeeded, _count);
        int dispensed   = notesUsed * _denomination;

        if (notesUsed > 0)
            Console.WriteLine($"Dispensing {notesUsed} x £{_denomination}");

        int remaining = amount - dispensed;
        return remaining > 0 ? (_next?.Dispense(remaining) ?? remaining) : 0;
    }
}

public class Fifty   : NoteDispenser { public Fifty()   { _denomination=50; _count=10; } }
public class Twenty  : NoteDispenser { public Twenty()  { _denomination=20; _count=20; } }
public class Ten     : NoteDispenser { public Ten()     { _denomination=10; _count=30; } }

var dispenser = new Fifty();
dispenser.SetNext(new Twenty()).SetNext(new Ten());

int unpaid = dispenser.Dispense(130);  // £50 + £50 + £20 + £10
Console.WriteLine(unpaid == 0 ? "✅ Done" : $"❌ Short by £{unpaid}");

// ── Domain Models ─────────────────────────────────────
public class Account
{
    public string  AccountId { get; }
    private decimal _balance;
    private readonly object _lock = new();

    public Account(string id, decimal initialBalance)
        => (AccountId, _balance) = (id, initialBalance);

    public decimal Balance { get { lock(_lock) return _balance; } }

    public bool Debit(decimal amount)
    {
        lock (_lock)
        {
            if (_balance < amount) return false;
            _balance -= amount;
            return true;
        }
    }

    public void Credit(decimal amount) { lock (_lock) _balance += amount; }
}

public class Card
{
    public string  CardNumber { get; }
    public Account Account    { get; }
    private readonly string _pinHash;

    public Card(string number, Account account, string pin)
        => (CardNumber, Account, _pinHash) = (number, account, Hash(pin));

    public bool ValidatePin(string pin) => Hash(pin) == _pinHash;

    private static string Hash(string pin) => pin.GetHashCode().ToString(); // simplified
}

// ── Cash Dispenser (Chain of Responsibility) ──────────
public class CashDispenser
{
    private readonly int _denomination;
    private int          _count;
    private CashDispenser? _next;

    public CashDispenser(int denomination, int count)
        => (_denomination, _count) = (denomination, count);

    public CashDispenser SetNext(CashDispenser next)
        => (_next = next, next).Item2;

    // Returns amount it couldn't dispense (0 = success)
    public int Dispense(int amount)
    {
        int used      = Math.Min(amount / _denomination, _count);
        int covered   = used * _denomination;
        _count       -= used;

        if (used > 0)
            Console.WriteLine($"  Dispensing {used} × £{_denomination}");

        int remaining = amount - covered;
        return remaining > 0 ? (_next?.Dispense(remaining) ?? remaining) : 0;
    }
}

// ── Idle State ────────────────────────────────────────
public class IdleState : IAtmState
{
    public void InsertCard(Atm atm, Card card)
    {
        atm.CurrentCard = card;
        Console.WriteLine($"Card {card.CardNumber} inserted. Please enter PIN.");
        atm.SetState(atm.CardInsertedState);
    }
    public void EnterPin(Atm a, string p)  => Console.WriteLine("Insert card first.");
    public void Withdraw(Atm a, decimal m) => Console.WriteLine("Insert card first.");
    public void Deposit(Atm a, decimal m)  => Console.WriteLine("Insert card first.");
    public void EjectCard(Atm a)           => Console.WriteLine("No card inserted.");
}

// ── Card Inserted State ───────────────────────────────
public class CardInsertedState : IAtmState
{
    private int _attempts = 0;

    public void InsertCard(Atm a, Card c) => Console.WriteLine("Card already inserted.");

    public void EnterPin(Atm atm, string pin)
    {
        if (atm.CurrentCard!.ValidatePin(pin))
        {
            _attempts = 0;
            Console.WriteLine("PIN accepted. Select transaction.");
            atm.SetState(atm.AuthenticatedState);
        }
        else
        {
            _attempts++;
            Console.WriteLine($"Wrong PIN. Attempt {_attempts}/3.");
            if (_attempts >= 3)
            {
                Console.WriteLine("Card blocked! Ejecting.");
                atm.EjectCard();
            }
        }
    }

    public void Withdraw(Atm a, decimal m) => Console.WriteLine("Enter PIN first.");
    public void Deposit(Atm a, decimal m)  => Console.WriteLine("Enter PIN first.");
    public void EjectCard(Atm atm)
    {
        atm.CurrentCard = null;
        Console.WriteLine("Card ejected.");
        atm.SetState(atm.IdleState);
    }
}

// ── Authenticated State ───────────────────────────────
public class AuthenticatedState : IAtmState
{
    public void InsertCard(Atm a, Card c) => Console.WriteLine("Session active.");

    public void EnterPin(Atm a, string p) => Console.WriteLine("Already authenticated.");

    public void Withdraw(Atm atm, decimal amount)
    {
        var account = atm.CurrentCard!.Account;
        Console.WriteLine($"Balance: £{account.Balance}. Withdrawing £{amount}...");

        if (!account.Debit(amount))
        { Console.WriteLine("Insufficient funds."); return; }

        int unpaid = atm.Dispenser.Dispense((int)amount);
        if (unpaid > 0)
        {
            // Refund if machine can't fully dispense
            account.Credit(unpaid);
            Console.WriteLine($"Could only dispense £{amount - unpaid}. Refunded £{unpaid}.");
        }
        else
            Console.WriteLine($"✅ Please collect £{amount}.");
    }

    public void Deposit(Atm atm, decimal amount)
    {
        atm.CurrentCard!.Account.Credit(amount);
        Console.WriteLine($"✅ Deposited £{amount}. New balance: £{atm.CurrentCard.Account.Balance}");
    }

    public void EjectCard(Atm atm)
    {
        Console.WriteLine("Session ended. Card ejected.");
        atm.CurrentCard = null;
        atm.SetState(atm.IdleState);
    }
}

// ── ATM Context ───────────────────────────────────────
public class Atm
{
    public IAtmState     IdleState          { get; } = new IdleState();
    public IAtmState     CardInsertedState  { get; } = new CardInsertedState();
    public IAtmState     AuthenticatedState { get; } = new AuthenticatedState();
    public CashDispenser  Dispenser          { get; }
    public Card?          CurrentCard        { get; set; }

    private IAtmState _state;

    public Atm()
    {
        _state = IdleState;
        // Build the cash dispenser chain
        var fifty   = new CashDispenser(50, 10);
        var twenty  = new CashDispenser(20, 20);
        var ten     = new CashDispenser(10, 30);
        var five    = new CashDispenser(5, 40);
        fifty.SetNext(twenty).SetNext(ten).SetNext(five);
        Dispenser = fifty;
    }

    public void SetState(IAtmState s) => _state = s;
    public void InsertCard(Card c)      => _state.InsertCard(this, c);
    public void EnterPin(string pin)    => _state.EnterPin(this, pin);
    public void Withdraw(decimal amt)   => _state.Withdraw(this, amt);
    public void Deposit(decimal amt)    => _state.Deposit(this, amt);
    public void EjectCard()             => _state.EjectCard(this);
}

Online Blackjack
Factory — Deck/Hands Template Method — Game Flow Strategy — Dealer AI
Key Design Decisions
  • Template Method: BlackjackGame.Play() defines the fixed flow — deal, player turns, dealer turn, settle. Subclasses can override steps
  • Ace value is dynamic — Hand.Value counts Aces as 11, then reduces if bust
  • Deck is a shuffled stack of cards; draw = pop
  • Dealer and Player both extend Participant — share hand logic
  • Result enum covers all outcomes: Win, Lose, Push, Blackjack, Bust
Game Flow (Template Method)
  • 1. DealInitialCards() — 2 to each participant
  • 2. CheckBlackjack() — instant win check
  • 3. PlayerTurns() — each player hits or stands
  • 4. DealerTurn() — dealer hits until ≥17
  • 5. SettleResults() — compare hands, pay out
Interviewer will ask: Q: How do you handle the Ace being 1 or 11? → In Hand.Value: count all Aces as 11 first. While total > 21 and there are Aces, reduce one Ace from 11 to 1 (subtract 10). This is the standard optimal Ace-counting algorithm. Q: What is the Template Method pattern doing here? → Play() in the base class calls abstract methods in a fixed order. A MultiDeckBlackjack subclass only overrides CreateDeck() — the game flow stays identical. The skeleton is locked, the steps are swappable.

📖 Pattern Primer Template Method — define the skeleton, let subclasses fill in steps

// ── Card & Deck ───────────────────────────────────────
public enum Suit  { Hearts, Diamonds, Clubs, Spades }
public enum Rank  { Two=2,Three,Four,Five,Six,Seven,Eight,Nine,Ten,Jack,Queen,King,Ace }

public record Card(Suit Suit, Rank Rank)
{
    // Face value — Ace handled in Hand.Value
    public int Value => Rank switch
    {
        Rank.Jack or Rank.Queen or Rank.King => 10,
        Rank.Ace                               => 11,
        _                                      => (int)Rank
    };
    public override string ToString() => $"{Rank} of {Suit}";
}

public class Deck
{
    private readonly Stack<Card> _cards;

    public Deck(int numDecks = 1)
    {
        var cards = new List<Card>();
        for (int d = 0; d < numDecks; d++)
            foreach (Suit s in Enum.GetValues<Suit>())
                foreach (Rank r in Enum.GetValues<Rank>())
                    cards.Add(new Card(s, r));

        // Shuffle using Fisher-Yates
        var rng = new Random();
        for (int i = cards.Count - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            (cards[i], cards[j]) = (cards[j], cards[i]);
        }
        _cards = new Stack<Card>(cards);
    }

    public Card Draw() => _cards.Pop();
    public int  Remaining => _cards.Count;
}

// ── Hand ──────────────────────────────────────────────
public class Hand
{
    private readonly List<Card> _cards = new();

    public void Add(Card card) => _cards.Add(card);
    public void Clear()       => _cards.Clear();

    // Ace = 11, reduced to 1 while bust — the standard algorithm
    public int Value
    {
        get
        {
            int total = _cards.Sum(c => c.Value);
            int aces  = _cards.Count(c => c.Rank == Rank.Ace);
            while (total > 21 && aces > 0) { total -= 10; aces--; }
            return total;
        }
    }

    public bool IsBust      => Value > 21;
    public bool IsBlackjack => _cards.Count == 2 && Value == 21;

    public override string ToString() =>
        string.Join(", ", _cards) + $" (Value: {Value})";
}

public abstract class Participant
{
    public string Name { get; }
    public Hand   Hand { get; } = new();

    protected Participant(string name) => Name = name;
    public abstract bool ShouldHit(Hand dealerHand);
}

public class Player : Participant
{
    public decimal Chips  { get; private set; }
    public decimal Bet    { get; private set; }

    public Player(string name, decimal chips) : base(name) => Chips = chips;

    public void PlaceBet(decimal amount)
    {
        if (amount > Chips) throw new InvalidOperationException("Insufficient chips.");
        Bet   = amount;
        Chips -= amount;
    }

    public void Win(decimal multiplier = 1m) => Chips += Bet * (1 + multiplier);
    public void Push()                        => Chips += Bet; // return bet

    // Human player hits if value is low — simplified (real: ask user)
    public override bool ShouldHit(Hand d) => Hand.Value < 17;
}

public class Dealer : Participant
{
    public Dealer() : base("Dealer") { }

    // Standard casino rule: dealer hits on soft 16 or less, stands on 17+
    public override bool ShouldHit(Hand _) => Hand.Value < 17;
}

// ── Blackjack Game (Template Method) ─────────────────
public class BlackjackGame
{
    private readonly List<Player> _players;
    private readonly Dealer       _dealer = new();
    private          Deck          _deck;

    public BlackjackGame(List<Player> players, int numDecks = 1)
    {
        _players = players;
        _deck    = new Deck(numDecks);
    }

    // Template Method — the fixed game skeleton
    public sealed void Play()
    {
        TakeBets();
        DealInitialCards();
        if (!CheckBlackjack())
        {
            PlayPlayerTurns();
            PlayDealerTurn();
        }
        SettleResults();
        ResetHands();
    }

    private void TakeBets()
    {
        foreach (var p in _players)
        {
            Console.Write($"{p.Name} (£{p.Chips} chips) — bet: ");
            p.PlaceBet(10m); // fixed £10 for simplicity
        }
    }

    private void DealInitialCards()
    {
        // Deal 2 cards to each player and dealer
        for (int i = 0; i < 2; i++)
        {
            foreach (var p in _players) p.Hand.Add(_deck.Draw());
            _dealer.Hand.Add(_deck.Draw());
        }
        foreach (var p in _players)
            Console.WriteLine($"{p.Name}: {p.Hand}");
        Console.WriteLine($"Dealer shows: {_dealer.Hand}");
    }

    // Returns true if someone has blackjack (skip player turns)
    private bool CheckBlackjack()
    {
        bool dealerBJ = _dealer.Hand.IsBlackjack;
        foreach (var p in _players)
        {
            if (p.Hand.IsBlackjack && dealerBJ) { Console.WriteLine($"{p.Name}: Push (both BJ)"); p.Push(); }
            else if (p.Hand.IsBlackjack)           { Console.WriteLine($"{p.Name}: Blackjack! 3:2"); p.Win(1.5m); }
            else if (dealerBJ)                        Console.WriteLine($"{p.Name}: Dealer Blackjack — you lose.");
        }
        return dealerBJ || _players.Any(p => p.Hand.IsBlackjack);
    }

    private void PlayPlayerTurns()
    {
        foreach (var p in _players)
        {
            Console.WriteLine($"\n{p.Name}'s turn:");
            while (!p.Hand.IsBust && p.ShouldHit(_dealer.Hand))
            {
                var card = _deck.Draw();
                p.Hand.Add(card);
                Console.WriteLine($"  Hit → {card}. Hand: {p.Hand}");
            }
            if (p.Hand.IsBust) Console.WriteLine($"  {p.Name} busts!");
        }
    }

    private void PlayDealerTurn()
    {
        Console.WriteLine($"\nDealer's turn. Hand: {_dealer.Hand}");
        while(_dealer.ShouldHit(_dealer.Hand))
        {
            var card = _deck.Draw();
            _dealer.Hand.Add(card);
            Console.WriteLine($"  Dealer hits → {card}. Hand: {_dealer.Hand}");
        }
    }

    private void SettleResults()
    {
        int dealerVal = _dealer.Hand.Value;
        foreach (var p in _players)
        {
            if      (p.Hand.IsBust)           Console.WriteLine($"{p.Name}: Bust — lose.");
            else if (_dealer.Hand.IsBust)     { Console.WriteLine($"{p.Name}: Dealer bust — win! Chips: £{p.Chips}"); p.Win(); }
            else if (p.Hand.Value > dealerVal) { Console.WriteLine($"{p.Name}: Win! Chips: £{p.Chips}"); p.Win(); }
            else if (p.Hand.Value < dealerVal)   Console.WriteLine($"{p.Name}: Lose.");
            else                               { Console.WriteLine($"{p.Name}: Push."); p.Push(); }
        }
    }

    private void ResetHands()
    {
        _players.ForEach(p => p.Hand.Clear());
        _dealer.Hand.Clear();
    }
}

📖 Pattern Primer Observer Pattern — subscribe to events, get notified automatically
One object (Subject/Publisher) maintains a list of dependents (Observers/Subscribers). When the subject's state changes, it notifies all observers. They don't poll — they get pushed. In .NET this maps neatly to events, but knowing the raw pattern is what interviewers want.

{

public class Library
{
    private static readonly Lazy<Library> _inst = new(() => new Library());
    public static Library Instance => _inst.Value;

    private readonly Dictionary<string, Book>     _catalogue = new();
    private readonly Dictionary<string, BookItem> _items     = new();
    private readonly Dictionary<string, Member>   _members   = new();
    private readonly IFineStrategy               _fine      = new DailyFine();
    private readonly INotificationService         _notify    = new EmailNotification();

    private Library() {}

    public void AddBook(Book book) => _catalogue[book.ISBN] = book;

    public void AddBookItem(BookItem item) => _items[item.Barcode] = item;

    public void RegisterMember(Member m) => _members[m.MemberId] = m;

    public Loan? Checkout(string memberId, string isbn)
    {
        if (!_members.TryGetValue(memberId, out var member) ||
             member.Status != MemberStatus.Active)
        { Console.WriteLine("Member not active."); return null; }

        // Find any available copy of this book
        var item = _items.Values
            .FirstOrDefault(i => i.Book.ISBN == isbn && i.Status == BookStatus.Available);

        if (item == null)
        {
            Console.WriteLine($"No copies available. Added {member.Name} to waitlist.");
            _catalogue[isbn].AddToWaitlist(member);
            return null;
        }

        item.Status = BookStatus.Borrowed;
        var loan   = new Loan(member, item);
        member.ActiveLoans.Add(loan);
        Console.WriteLine($"✅ {member.Name} borrowed '{item.Book.Title}'. Due: {loan.DueDate:dd MMM}");
        return loan;
    }

    public decimal Return(Loan loan)
    {
        loan.ReturnedOn = DateTime.Today;
        decimal fine    = _fine.Calculate(loan.DueDate, DateTime.Today);
        loan.Fine       = fine;

        loan.BookItem.Status = BookStatus.Available;
        loan.Member.ActiveLoans.Remove(loan);

        if (fine > 0)
            Console.WriteLine($"📕 Returned with £{fine} fine.");
        else
            Console.WriteLine($"📗 Returned on time.");

        // Observer: notify the next member on the waitlist
        loan.BookItem.Book.NotifyNextInQueue(_notify);
        return fine;
    }

    public IEnumerable<Book> Search(string query) =>
        _catalogue.Values.Where(b =>
            b.Title.Contains(query, StringComparison.OrdinalIgnoreCase) ||
            b.Author.Contains(query, StringComparison.OrdinalIgnoreCase) ||
            b.ISBN == query);
}

public enum Polarity  { Tab, Blank, Flat }    // Flat = border edge
public enum EdgeSide  { Top, Right, Bottom, Left }

public record PieceEdge(EdgeSide Side, int ShapeId, Polarity Polarity)
{
    // Two edges are compatible if they share a shape but oppose in polarity
    public bool FitsOpposite(PieceEdge other) =>
        ShapeId == other.ShapeId &&
        ((Polarity == Polarity.Tab   && other.Polarity == Polarity.Blank) ||
         (Polarity == Polarity.Blank && other.Polarity == Polarity.Tab));
}

// ── Puzzle Board ──────────────────────────────────────
public class PuzzleBoard
{
    public int Rows { get; }
    public int Cols { get; }
    private readonly Piece?[,] _grid;

    public int PlacedCount  { get; private set; }
    public int TotalPieces  => Rows * Cols;
    public bool IsComplete  => PlacedCount == TotalPieces;

    public PuzzleBoard(int rows, int cols)
    {
        Rows = rows; Cols = cols;
        _grid = new Piece[rows, cols];
    }

    public bool TryPlace(Piece piece, int row, int col)
    {
        if (row < 0 || row >= Rows || col < 0 || col >= Cols)
            return false;
        if (_grid[row, col] != null)
        { Console.WriteLine($"Position ({row},{col}) already occupied."); return false; }

        if (!ValidateNeighbours(piece, row, col))
        { Console.WriteLine($"Piece {piece.Id} doesn't fit at ({row},{col})."); return false; }

        _grid[row, col] = piece;
        piece.IsPlaced = true;
        PlacedCount++;
        return true;
    }

    public bool RemovePiece(int row, int col)
    {
        var piece = _grid[row, col];
        if (piece == null) return false;
        _grid[row, col]  = null;
        piece.IsPlaced  = false;
        PlacedCount--;
        return true;
    }

    private bool ValidateNeighbours(Piece piece, int row, int col)
    {
        // Check top neighbour
        if (row > 0 && _grid[row-1, col] != null)
        {
            var nb = _grid[row-1, col]!;
            if (!piece.GetEdge(EdgeSide.Top).FitsOpposite(nb.GetEdge(EdgeSide.Bottom)))
                return false;
        }
        // Check left neighbour
        if (col > 0 && _grid[row, col-1] != null)
        {
            var nb = _grid[row, col-1]!;
            if (!piece.GetEdge(EdgeSide.Left).FitsOpposite(nb.GetEdge(EdgeSide.Right)))
                return false;
        }
        return true;
    }

    public void Print()
    {
        for (int r = 0; r < Rows; r++)
        {
            for (int c = 0; c < Cols; c++)
                Console.Write(_grid[r,c] == null ? "[ ]" : $"[{_grid[r,c]!.Id,2}]");
            Console.WriteLine();
        }
    }
}

// ── Puzzle — ties everything together ─────────────────
public class Puzzle
{
    public PuzzleBoard  Board  { get; }
    public List<Piece>  Pieces { get; }

    private readonly Stack<ICommand> _history = new();

    // Observer: fire when puzzle is complete
    public event Action? OnCompleted;

    public Puzzle(PuzzleBoard board, int pieceCount)
    {
        Board  = board;
        Pieces = GeneratePieces(pieceCount, board.Rows, board.Cols);
    }

    public bool PlacePiece(Piece piece, int row, int col)
    {
        var cmd = new PlacePieceCommand(Board, piece, row, col);
        if (!cmd.Execute()) return false;

        _history.Push(cmd);
        Console.WriteLine($"✅ Placed piece {piece.Id} at ({row},{col}). Progress: {Board.PlacedCount}/{Board.TotalPieces}");

        if (Board.IsComplete)
        {
            Console.WriteLine("🎉 Puzzle complete!");
            OnCompleted?.Invoke();
        }
        return true;
    }

    public void UndoLastPlace()
    {
        if (_history.Count > 0)
            _history.Pop().Undo();
    }

    // Factory: generate pieces with compatible edge shapes
    private static List<Piece> GeneratePieces(int count, int rows, int cols)
    {
        var pieces = new List<Piece>();
        var rng    = new Random(42); // seed for reproducibility
        int shapeCounter = 1;

        // Build horizontal and vertical edge maps so adjacent pieces match
        var hEdges = new int[rows + 1, cols]; // horizontal borders
        var vEdges = new int[rows, cols + 1]; // vertical borders

        for (int r = 0; r <= rows; r++)
            for (int c = 0; c < cols; c++)
                hEdges[r, c] = shapeCounter++;

        for (int r = 0; r < rows; r++)
            for (int c = 0; c <= cols; c++)
                vEdges[r, c] = shapeCounter++;

        int id = 1;
        for (int r = 0; r < rows; r++)
        {
            for (int c = 0; c < cols; c++)
            {
                var piece = new Piece(id++);

                // Top edge: Flat if border row, otherwise Tab/Blank paired with neighbour above
                piece.Edges[EdgeSide.Top]    = new PieceEdge(EdgeSide.Top,    hEdges[r, c],   r == 0        ? Polarity.Flat : Polarity.Blank);
                piece.Edges[EdgeSide.Bottom] = new PieceEdge(EdgeSide.Bottom, hEdges[r+1, c], r == rows - 1 ? Polarity.Flat : Polarity.Tab);
                piece.Edges[EdgeSide.Left]   = new PieceEdge(EdgeSide.Left,   vEdges[r, c],   c == 0        ? Polarity.Flat : Polarity.Blank);
                piece.Edges[EdgeSide.Right]  = new PieceEdge(EdgeSide.Right,  vEdges[r, c+1], c == cols - 1 ? Polarity.Flat : Polarity.Tab);

                pieces.Add(piece);
            }
        }
        return pieces;
    }
}
