Tic Tac Toe
State Pattern Factory Strategy (AI)
Key Design Decisions
  • Board is a 2D array of CellState enum — simple and clear
  • Game owns state; Board is just data + win detection
  • IPlayer interface lets Human and AI be swappable (Strategy)
  • Game state (Active / Won / Draw) lives in an enum — no booleans
Key Classes
  • Board — holds cells, checks win/draw
  • Game — orchestrates turns, switches players
  • IPlayer — Human or AI, both make a Move
  • Move — a row/col value object
  • GameStatus enum — InProgress / Won / Draw
Interviewer will ask: Q: How would you extend this to a 4×4 or NxN board? → Board takes int size in constructor. Win check uses size as threshold. Zero code change elsewhere. Q: How do you add an AI player? → Implement IPlayer with a minimax strategy. Game doesn't change at all — that's the Strategy pattern.
Patterns: State — GameStatus Strategy — IPlayer Factory — PlayerFactory
📖 Pattern Primer Strategy Pattern — swap algorithms without changing the caller
Define an interface for a behaviour. Write multiple implementations. The caller holds the interface — it never needs to change when you add a new strategy.
// ✅ Strategy: extract algorithm behind an interface
public interface ISortStrategy
{
    void Sort(int[] data);
}

public class BubbleSort : ISortStrategy
{
    public void Sort(int[] data) => Console.WriteLine("Bubble sorting...");
}

public class QuickSort : ISortStrategy
{
    public void Sort(int[] data) => Console.WriteLine("Quick sorting...");
}

public class Sorter
{
    private ISortStrategy _strategy;
    public Sorter(ISortStrategy strategy) => _strategy = strategy;
    public void SetStrategy(ISortStrategy s) => _strategy = s;
    public void Sort(int[] data) => _strategy.Sort(data);
}

// Caller swaps strategies — Sorter never changes
var sorter = new Sorter(new BubbleSort());
sorter.Sort(new[] { 3, 1, 2 });        // Bubble sorting...
sorter.SetStrategy(new QuickSort());
sorter.Sort(new[] { 3, 1, 2 });        // Quick sorting...

TicTacToe.cs
C# 10
// ── Enums ─────────────────────────────────────────────
public enum CellState { Empty, X, O }

public enum GameStatus { InProgress, Won, Draw }

// ── Value Object ──────────────────────────────────────
public record Move(int Row, int Col);

// ── Board ─────────────────────────────────────────────
public class Board
{
    private readonly CellState[,] _cells;
    public int Size { get; }

    public Board(int size = 3)
    {
        Size = size;
        _cells = new CellState[size, size];
    }

    public bool IsCellEmpty(int row, int col) =>
        _cells[row, col] == CellState.Empty;

    public void PlaceMark(int row, int col, CellState mark)
    {
        if (!IsCellEmpty(row, col))
            throw new InvalidOperationException("Cell already taken.");
        _cells[row, col] = mark;
    }

    public bool HasWinner(CellState mark)
    {
        // Check all rows and columns
        for (int i = 0; i < Size; i++)
        {
            if (AllMatch(mark, Enumerable.Range(0, Size).Select(j => (i, j)))) return true;
            if (AllMatch(mark, Enumerable.Range(0, Size).Select(j => (j, i)))) return true;
        }
        // Check diagonals
        if (AllMatch(mark, Enumerable.Range(0, Size).Select(i => (i, i)))) return true;
        if (AllMatch(mark, Enumerable.Range(0, Size).Select(i => (i, Size - 1 - i)))) return true;
        return false;
    }

    public bool IsFull => Enumerable.Range(0, Size)
        .SelectMany(r => Enumerable.Range(0, Size).Select(c => (r, c)))
        .All(p => _cells[p.r, p.c] != CellState.Empty);

    private bool AllMatch(CellState mark, IEnumerable<(int, int)> cells) =>
        cells.All(p => _cells[p.Item1, p.Item2] == mark);

    public void Print()
    {
        for (int r = 0; r < Size; r++)
        {
            for (int c = 0; c < Size; c++)
                Console.Write($"{(_cells[r, c] == CellState.Empty ? "." : _cells[r, c].ToString()),2}");
            Console.WriteLine();
        }
    }
}

// ── Player (Strategy Pattern) ─────────────────────────
public interface IPlayer
{
    CellState Mark { get; }
    Move GetMove(Board board);
}

public class HumanPlayer : IPlayer
{
    public CellState Mark { get; }
    public HumanPlayer(CellState mark) => Mark = mark;

    public Move GetMove(Board board)
    {
        Console.Write($"Player {Mark} — enter row,col: ");
        var parts = Console.ReadLine()!.Split(',');
        return new Move(int.Parse(parts[0]), int.Parse(parts[1]));
    }
}

// AI picks first available empty cell — swap for minimax for a real AI
public class AiPlayer : IPlayer
{
    public CellState Mark { get; }
    public AiPlayer(CellState mark) => Mark = mark;

    public Move GetMove(Board board)
    {
        for (int r = 0; r < board.Size; r++)
            for (int c = 0; c < board.Size; c++)
                if (board.IsCellEmpty(r, c)) return new Move(r, c);
        throw new InvalidOperationException("No moves left.");
    }
}

// ── Game Orchestrator ─────────────────────────────────
public class Game
{
    private readonly Board _board;
    private readonly IPlayer[] _players;
    private int _currentIndex = 0;

    public GameStatus Status { get; private set; } = GameStatus.InProgress;
    public IPlayer? Winner { get; private set; }

    public Game(IPlayer p1, IPlayer p2, int boardSize = 3)
    {
        _board = new Board(boardSize);
        _players = new[] { p1, p2 };
    }

    public void PlayTurn()
    {
        if (Status != GameStatus.InProgress) return;

        var player = _players[_currentIndex];
        var move   = player.GetMove(_board);

        _board.PlaceMark(move.Row, move.Col, player.Mark);
        _board.Print();

        if (_board.HasWinner(player.Mark))
        {
            Status = GameStatus.Won;
            Winner = player;
        }
        else if (_board.IsFull)
        {
            Status = GameStatus.Draw;
        }

        // Switch player
        _currentIndex = (_currentIndex + 1) % _players.Length;
    }

    public void Run()
    {
        while (Status == GameStatus.InProgress)
            PlayTurn();

        Console.WriteLine(Status == GameStatus.Won
            ? $"🎉 {Winner!.Mark} wins!"
            : "🤝 It's a draw!");
    }
}

// ── Entry Point ───────────────────────────────────────
var game = new Game(
    new HumanPlayer(CellState.X),
    new AiPlayer(CellState.O)
);
game.Run();

Snake & Food
State Pattern Observer Game Loop
Key Design Decisions
  • LinkedList<Point> for the snake body — O(1) add head, remove tail
  • HashSet<Point> for O(1) self-collision detection
  • GameEngine is a Singleton — owns the game loop
  • Direction changes are buffered — prevents reversing into self
  • Food spawning is a Strategy — easy to change algorithm
Key Classes
  • Point — (Row, Col) value object
  • Snake — body as LinkedList + HashSet
  • Board — width/height, food position
  • GameEngine — game loop, collision detection
  • GameState — Running / Paused / GameOver
Interviewer will ask: Q: Why LinkedList for the snake body? → AddFirst (new head) and RemoveLast (old tail) are both O(1). A List would be O(n) for removal from front. Q: How do you handle self-collision efficiently? → Maintain a parallel HashSet of occupied points. Checking if the new head position is in the set is O(1).
Patterns: State — GameState enum Singleton — GameEngine Strategy — Food spawner
📖 Pattern Primer Singleton Pattern — one instance, globally accessible
Guarantee that a class has exactly one instance. In .NET, use Lazy<T> — it's thread-safe and handles double-checked locking for you automatically.
// ❌ Without Singleton: anyone can create multiple instances
var db1 = new Database();  // separate connection
var db2 = new Database();  // another — wasteful!

// ✅ Singleton with Lazy<T> — the .NET idiomatic way
public class Database
{
    // Lazy ensures creation is deferred AND thread-safe
    private static readonly Lazy<Database> _instance =
        new(() => new Database());

    // Private constructor — no one else can call new Database()
    private Database() { Console.WriteLine("DB connected."); }

    public static Database Instance => _instance.Value;

    public void Query(string sql) => Console.WriteLine($"Running: {sql}");
}

// Both variables point to the exact same object
var a = Database.Instance;
var b = Database.Instance;
Console.WriteLine(ReferenceEquals(a, b));  // True

// In Snake → GameEngine.Instance is the one-and-only game loop controller.
SnakeGame.cs
C# 10
// ── Enums & Value Objects ─────────────────────────────
public enum Direction { Up, Down, Left, Right }
public enum GameState  { Running, Paused, GameOver }

public record Point(int Row, int Col);

// ── Snake ─────────────────────────────────────────────
public class Snake
{
    // LinkedList for O(1) head insert and tail removal
    private readonly LinkedList<Point> _body = new();
    // HashSet for O(1) collision checks
    private readonly HashSet<Point>   _occupied = new();

    public Point Head    => _body.First!.Value;
    public int   Length  => _body.Count;

    public Snake(Point start)
    {
        _body.AddFirst(start);
        _occupied.Add(start);
    }

    // Returns the tail point that was removed (null if snake grew)
    public Point? Move(Point newHead, bool grow)
    {
        _body.AddFirst(newHead);
        _occupied.Add(newHead);

        if (grow) return null;

        var tail = _body.Last!.Value;
        _body.RemoveLast();
        _occupied.Remove(tail);
        return tail;
    }

    public bool Occupies(Point p) => _occupied.Contains(p);
    public IEnumerable<Point> Body => _body;
}

// ── Food Spawner (Strategy) ───────────────────────────
public interface IFoodSpawner
{
    Point Spawn(int rows, int cols, IEnumerable<Point> occupied);
}

public class RandomFoodSpawner : IFoodSpawner
{
    private readonly Random _rng = new();

    public Point Spawn(int rows, int cols, IEnumerable<Point> occupied)
    {
        var taken = new HashSet<Point>(occupied);
        Point candidate;
        do { candidate = new Point(_rng.Next(rows), _rng.Next(cols)); }
        while (taken.Contains(candidate));
        return candidate;
    }
}

// ── Board ─────────────────────────────────────────────
public class Board
{
    public int Rows { get; }
    public int Cols { get; }
    public Point Food { get; private set; }

    private readonly IFoodSpawner _spawner;

    public Board(int rows, int cols, IFoodSpawner? spawner = null)
    {
        Rows = rows; Cols = cols;
        _spawner = spawner ?? new RandomFoodSpawner();
        Food = new Point(rows / 2, cols / 2); // initial food position
    }

    public bool IsInBounds(Point p) =>
        p.Row >= 0 && p.Row < Rows && p.Col >= 0 && p.Col < Cols;

    public void SpawnFood(IEnumerable<Point> snakeBody) =>
        Food = _spawner.Spawn(Rows, Cols, snakeBody);
}

// ── Game Engine (Singleton) ───────────────────────────
public class GameEngine
{
    private static readonly Lazy<GameEngine> _instance =
        new(() => new GameEngine());
    public static GameEngine Instance => _instance.Value;

    private GameEngine() { }

    private Snake     _snake    = null!;
    private Board     _board    = null!;
    private Direction _direction = Direction.Right;

    public GameState State { get; private set; } = GameState.GameOver;
    public int       Score { get; private set; }

    public void Start(int rows = 10, int cols = 10)
    {
        _board     = new Board(rows, cols);
        _snake     = new Snake(new Point(rows / 2, 0));
        _direction = Direction.Right;
        Score      = 0;
        State      = GameState.Running;
    }

    // Called by UI/input handler on keypress
    public void ChangeDirection(Direction newDir)
    {
        // Prevent 180-degree reversal
        bool invalid =
            (newDir == Direction.Up    && _direction == Direction.Down)  ||
            (newDir == Direction.Down  && _direction == Direction.Up)    ||
            (newDir == Direction.Left  && _direction == Direction.Right) ||
            (newDir == Direction.Right && _direction == Direction.Left);

        if (!invalid) _direction = newDir;
    }

    // Called once per game tick
    public void Tick()
    {
        if (State != GameState.Running) return;

        var newHead = NextHead(_snake.Head, _direction);

        // Wall collision
        if (!_board.IsInBounds(newHead)) { State = GameState.GameOver; return; }
        // Self collision
        if (_snake.Occupies(newHead))    { State = GameState.GameOver; return; }

        bool ateFood = newHead == _board.Food;
        _snake.Move(newHead, grow: ateFood);

        if (ateFood)
        {
            Score++;
            _board.SpawnFood(_snake.Body);
        }
    }

    private static Point NextHead(Point head, Direction dir) => dir switch
    {
        Direction.Up    => head with { Row = head.Row - 1 },
        Direction.Down  => head with { Row = head.Row + 1 },
        Direction.Left  => head with { Col = head.Col - 1 },
        Direction.Right => head with { Col = head.Col + 1 },
        _ => throw new ArgumentOutOfRangeException()
    };
}

Vending Machine
State Pattern ★ Singleton Strategy — Pricing
Key Design Decisions
  • State Pattern is the core — the machine behaves differently in each state
  • Each IVendingState only allows valid actions — invalid ones throw
  • VendingMachine context delegates all behaviour to the current state
  • Inventory is a simple Dictionary<string, int> — keep it simple
States
  • IdleState — waiting for coin insert
  • HasMoneyState — coin inserted, awaiting product select
  • DispensingState — dispensing product
  • OutOfStockState — machine has no inventory
Interviewer will ask: Q: Why use State pattern and not just if/else? → With 4 states and 4 actions you'd have 16 if/else branches. Adding a 5th state means touching every action method. State pattern keeps each state self-contained — Open/Closed Principle. Q: What if someone inserts more money than the item costs? → Track _insertedAmount in the context. After dispensing, return change: insertedAmount - itemPrice.
Patterns: State ★ — core of the design Singleton — VendingMachine
📖 Pattern Primer State Pattern — object behaviour changes with its internal state
Instead of giant if/else chains checking current state inside every method, each state is its own class that implements the same interface. The context just delegates to whichever state is current.
// Imagine a TrafficLight. Without State Pattern:
// every method has: if(state == Red) ... else if(state == Green) ...
// That's O(states × actions) if/else blocks.

// ✅ State Pattern
public interface ITrafficLightState
{
    void Handle(TrafficLight light);
}

public class RedState : ITrafficLightState
{
    public void Handle(TrafficLight light)
    {
        Console.WriteLine("RED — Stop.");
        light.SetState(new GreenState());  // transitions to next state
    }
}

public class GreenState : ITrafficLightState
{
    public void Handle(TrafficLight light)
    {
        Console.WriteLine("GREEN — Go.");
        light.SetState(new RedState());
    }
}

public class TrafficLight
{
    private ITrafficLightState _state = new RedState();
    public void SetState(ITrafficLightState s) => _state = s;
    public void Change() => _state.Handle(this); // delegate — no if/else here
}

var light = new TrafficLight();
light.Change();  // RED — Stop.  → transitions to Green
light.Change();  // GREEN — Go.  → transitions to Red
light.Change();  // RED — Stop.

// In VendingMachine → Idle / HasMoney / Dispensing / OutOfStock
// are all IVendingState implementations. Machine.InsertMoney() just
// calls _currentState.InsertMoney(this) — zero if/else in the machine.
VendingMachine.cs
C# 10
// ── Product ───────────────────────────────────────────
public record Product(string Name, decimal Price);

// ── State Interface ───────────────────────────────────
public interface IVendingState
{
    void InsertMoney(VendingMachine machine, decimal amount);
    void SelectProduct(VendingMachine machine, string productName);
    void Dispense(VendingMachine machine);
    void Cancel(VendingMachine machine);
}

// ── Idle State ────────────────────────────────────────
public class IdleState : IVendingState
{
    public void InsertMoney(VendingMachine m, decimal amount)
    {
        m.InsertedAmount += amount;
        Console.WriteLine($"Inserted ${amount}. Total: ${m.InsertedAmount}");
        m.SetState(m.HasMoneyState);
    }

    public void SelectProduct(VendingMachine m, string _) =>
        Console.WriteLine("Please insert money first.");

    public void Dispense(VendingMachine m) =>
        Console.WriteLine("Please insert money and select a product.");

    public void Cancel(VendingMachine m) =>
        Console.WriteLine("Nothing to cancel.");
}

// ── Has Money State ───────────────────────────────────
public class HasMoneyState : IVendingState
{
    public void InsertMoney(VendingMachine m, decimal amount)
    {
        m.InsertedAmount += amount;
        Console.WriteLine($"Added ${amount}. Total: ${m.InsertedAmount}");
    }

    public void SelectProduct(VendingMachine m, string name)
    {
        if (!m.Inventory.TryGetValue(name, out var stock) || stock == 0)
        { Console.WriteLine($"'{name}' is out of stock."); return; }

        if (!m.Products.TryGetValue(name, out var product))
        { Console.WriteLine($"Unknown product."); return; }

        if (m.InsertedAmount < product.Price)
        { Console.WriteLine($"Need ${product.Price - m.InsertedAmount} more."); return; }

        m.SelectedProduct = product;
        m.SetState(m.DispensingState);
        m.Dispense();  // auto-trigger dispense
    }

    public void Dispense(VendingMachine m) =>
        Console.WriteLine("Please select a product first.");

    public void Cancel(VendingMachine m)
    {
        Console.WriteLine($"Returned ${m.InsertedAmount}.");
        m.InsertedAmount = 0;
        m.SetState(m.IdleState);
    }
}

// ── Dispensing State ──────────────────────────────────
public class DispensingState : IVendingState
{
    public void InsertMoney(VendingMachine m, decimal _) =>
        Console.WriteLine("Dispensing in progress, please wait.");

    public void SelectProduct(VendingMachine m, string _) =>
        Console.WriteLine("Dispensing in progress, please wait.");

    public void Dispense(VendingMachine m)
    {
        var p = m.SelectedProduct!;
        m.Inventory[p.Name]--;
        decimal change = m.InsertedAmount - p.Price;

        Console.WriteLine($"🎉 Dispensing {p.Name}!");
        if (change > 0) Console.WriteLine($"Change: ${change}");

        m.InsertedAmount = 0;
        m.SelectedProduct = null;

        // Go to OutOfStock if inventory now zero, else Idle
        bool anyStock = m.Inventory.Values.Any(v => v > 0);
        m.SetState(anyStock ? m.IdleState : m.OutOfStockState);
    }

    public void Cancel(VendingMachine m) =>
        Console.WriteLine("Cannot cancel while dispensing.");
}

// ── Out-Of-Stock State ────────────────────────────────
public class OutOfStockState : IVendingState
{
    public void InsertMoney(VendingMachine m, decimal _) =>
        Console.WriteLine("Machine is out of stock. Money not accepted.");
    public void SelectProduct(VendingMachine m, string _) =>
        Console.WriteLine("Machine is out of stock.");
    public void Dispense(VendingMachine m) =>
        Console.WriteLine("Machine is out of stock.");
    public void Cancel(VendingMachine m) =>
        Console.WriteLine("Nothing to cancel.");
}

// ── Vending Machine Context (Singleton) ───────────────
public class VendingMachine
{
    private static readonly Lazy<VendingMachine> _instance =
        new(() => new VendingMachine());
    public static VendingMachine Instance => _instance.Value;

    // Pre-create all states — avoids allocation on every transition
    public IVendingState IdleState       { get; } = new IdleState();
    public IVendingState HasMoneyState   { get; } = new HasMoneyState();
    public IVendingState DispensingState { get; } = new DispensingState();
    public IVendingState OutOfStockState { get; } = new OutOfStockState();

    private IVendingState _currentState;

    public Dictionary<string, Product> Products  { get; } = new();
    public Dictionary<string, int>     Inventory { get; } = new();
    public decimal                       InsertedAmount  { get; set; }
    public Product?                      SelectedProduct { get; set; }

    private VendingMachine() => _currentState = IdleState;

    public void LoadProduct(Product p, int qty)
    {
        Products[p.Name]  = p;
        Inventory[p.Name] = qty;
        if (_currentState is OutOfStockState) SetState(IdleState);
    }

    // Delegate all actions to current state
    public void InsertMoney(decimal amount)    => _currentState.InsertMoney(this, amount);
    public void SelectProduct(string name)      => _currentState.SelectProduct(this, name);
    public void Dispense()                      => _currentState.Dispense(this);
    public void Cancel()                         => _currentState.Cancel(this);
    public void SetState(IVendingState state)   => _currentState = state;
}

// ── Usage ─────────────────────────────────────────────
var vm = VendingMachine.Instance;
vm.LoadProduct(new Product("Cola", 1.50m), qty: 5);
vm.LoadProduct(new Product("Chips", 1.00m), qty: 3);

vm.InsertMoney(1.50m);    // → HasMoneyState
vm.SelectProduct("Cola"); // → DispensingState → Idle
vm.InsertMoney(0.50m);    // need more money message
vm.SelectProduct("Chips");// need more money
vm.InsertMoney(0.50m);    // total $1.00
vm.SelectProduct("Chips");// → dispenses, no change

Logging System
Chain of Responsibility ★ Singleton Strategy — Formatters
Key Design Decisions
  • Chain of Responsibility: each handler checks if it should handle the log level, then passes it on
  • Logger is a Singleton — one entry point system-wide
  • Handlers are composable — add FileHandler, DbHandler without changing Logger
  • Log level is an int — trivial to compare (DEBUG=0, INFO=1, etc.)
Key Classes
  • LogLevel enum — Debug / Info / Warning / Error
  • LogMessage — level + message + timestamp
  • LogHandler — abstract base, has a Next handler
  • ConsoleHandler / FileHandler — concrete handlers
  • Logger — Singleton, builds and holds the chain
Interviewer will ask: Q: Why Chain of Responsibility over a simple switch? → Adding a new handler (e.g. SlackHandler for errors) requires zero changes to Logger or existing handlers. Open/Closed. With a switch, you'd touch the Logger every time. Q: How would you make this thread-safe? → Use a Channel<LogMessage> as a buffer. A single background Task drains the channel and calls the chain — serialises all writes, no locks needed on handlers.
Patterns: Chain of Responsibility ★ Singleton — Logger Decorator — formatters
📖 Pattern Primer Chain of Responsibility — pass a request along a chain of handlers
Each handler decides: "can I handle this? If yes, handle it. Then pass it along regardless." Handlers are linked together. The sender just calls the first one and forgets about the rest.
// Think of customer support tiers:
// Level1 handles basic issues → escalates → Level2 → Level3

public abstract class SupportHandler
{
    private SupportHandler? _next;

    // SetNext returns next so you can chain: h1.SetNext(h2).SetNext(h3)
    public SupportHandler SetNext(SupportHandler next)
    {
        _next = next;
        return next;
    }

    public void Handle(int severity)
    {
        if (CanHandle(severity))
            Console.WriteLine($"{GetType().Name} handled severity {severity}");

        _next?.Handle(severity);  // always pass along
    }

    protected abstract bool CanHandle(int severity);
}

public class Level1Support : SupportHandler
{
    protected override bool CanHandle(int s) => s <= 1;
}
public class Level2Support : SupportHandler
{
    protected override bool CanHandle(int s) => s <= 3;
}
public class Level3Support : SupportHandler
{
    protected override bool CanHandle(int s) => true;
}

// Build the chain
var l1 = new Level1Support();
l1.SetNext(new Level2Support()).SetNext(new Level3Support());

l1.Handle(1);  // Level1 handles it, passes to Level2 (can't), passes to Level3 (can)
l1.Handle(5);  // Level1 skips, Level2 skips, Level3 handles

// In Logger → ConsoleHandler (Debug+), FileHandler (Warning+),
// AlertHandler (Error only). Logger calls chain head — done.
LoggingSystem.cs
C# 10
// ── Log Level & Message ───────────────────────────────
public enum LogLevel { Debug = 0, Info = 1, Warning = 2, Error = 3 }

public record LogMessage(
    LogLevel  Level,
    string    Message,
    DateTime  Timestamp)
{
    public override string ToString() =>
        $"[{Timestamp:HH:mm:ss}] [{Level,-7}] {Message}";
}

// ── Abstract Handler (Chain of Responsibility) ────────
public abstract class LogHandler
{
    private LogHandler? _next;

    // Fluent builder: ConsoleHandler.SetNext(FileHandler).SetNext(AlertHandler)
    public LogHandler SetNext(LogHandler next)
    {
        _next = next;
        return next; // return next so calls can be chained
    }

    public void Handle(LogMessage msg)
    {
        if (CanHandle(msg))
            Write(msg);

        // Always pass down the chain regardless
        _next?.Handle(msg);
    }

    protected abstract bool CanHandle(LogMessage msg);
    protected abstract void Write(LogMessage msg);
}

// ── Console Handler ───────────────────────────────────
public class ConsoleLogHandler : LogHandler
{
    private readonly LogLevel _minLevel;

    public ConsoleLogHandler(LogLevel minLevel = LogLevel.Debug)
        => _minLevel = minLevel;

    protected override bool CanHandle(LogMessage msg) =>
        msg.Level >= _minLevel;

    protected override void Write(LogMessage msg)
    {
        Console.ForegroundColor = msg.Level switch
        {
            LogLevel.Error   => ConsoleColor.Red,
            LogLevel.Warning => ConsoleColor.Yellow,
            LogLevel.Info    => ConsoleColor.Cyan,
            _                => ConsoleColor.Gray
        };
        Console.WriteLine(msg);
        Console.ResetColor();
    }
}

        msg.Level >= _minLevel;

{
    protected override bool CanHandle(LogMessage msg) =>

    protected override void Write(LogMessage msg) =>
        Console.WriteLine($"🚨 ALERT: {msg.Message} — would send email/SMS here");
}

// ── Logger Singleton ──────────────────────────────────
public class Logger
{
    private static readonly Lazy<Logger> _instance = new(() => new Logger());
    public static Logger Instance => _instance.Value;

    private readonly LogHandler _chain;

    private Logger()
    {
        // Build the chain: Console → File → Alert
        var console = new ConsoleLogHandler(LogLevel.Debug);
        var file    = new FileLogHandler("app.log", LogLevel.Warning);
        var alert   = new AlertLogHandler();

        console.SetNext(file).SetNext(alert);
        _chain = console;
    }

    public void Log(LogLevel level, string message) =>
        _chain.Handle(new LogMessage(level, message, DateTime.UtcNow));

    // Convenience methods
    public void Debug(string msg)   => Log(LogLevel.Debug,   msg);
    public void Info(string msg)    => Log(LogLevel.Info,    msg);
    public void Warning(string msg) => Log(LogLevel.Warning, msg);
    public void Error(string msg)   => Log(LogLevel.Error,   msg);
}

}

{

    private List<IFileSystemItem> _children = new();

    {
        Console.WriteLine($"{indent}📁 {Name}");
        foreach (var child in _children)
            child.Print(indent + "  ");
    }
}

// ── Ticket (Value Object) ─────────────────────────────
public class Ticket
{
    public Guid         Id         { get; } = Guid.NewGuid();
    public Vehicle      Vehicle    { get; }
    public ParkingSpot  Spot       { get; }
    public DateTime     EntryTime  { get; } = DateTime.UtcNow;
    public DateTime?    ExitTime   { get; private set; }
    public decimal?    Fee        { get; private set; }

    public Ticket(Vehicle vehicle, ParkingSpot spot)
    {
        Vehicle = vehicle;
        Spot    = spot;
    }

    public void Settle(decimal fee)
    {
        ExitTime = DateTime.UtcNow;
        Fee      = fee;
    }

    public TimeSpan Duration =>
        (ExitTime ?? DateTime.UtcNow) - EntryTime;
}

// ── Parking Spot ──────────────────────────────────────
public class ParkingSpot
{
    public int         Id     { get; }
    public SpotType   Type   { get; }
    public SpotStatus Status { get; private set; } = SpotStatus.Available;

    public ParkingSpot(int id, SpotType type) { Id = id; Type = type; }

    public bool IsAvailable => Status == SpotStatus.Available;

    public void Occupy()   => Status = SpotStatus.Occupied;
    public void Vacate()   => Status = SpotStatus.Available;
}

// ── Pricing Strategy ──────────────────────────────────
public interface IPricingStrategy
{
    decimal Calculate(Ticket ticket);
}

public class HourlyPricing : IPricingStrategy
{
    private readonly decimal _ratePerHour;
    public HourlyPricing(decimal ratePerHour = 2.50m) => _ratePerHour = ratePerHour;

    public decimal Calculate(Ticket ticket)
    {
        var hours = (decimal)Math.Ceiling(ticket.Duration.TotalHours);
        return hours * _ratePerHour;
    }
}

// ── Parking Floor (Composite leaf) ────────────────────
public class ParkingFloor
{
    public int FloorNumber { get; }
    private readonly List<ParkingSpot> _spots = new();

    public ParkingFloor(int floorNumber) => FloorNumber = floorNumber;

    public void AddSpot(ParkingSpot spot) => _spots.Add(spot);

    public ParkingSpot? FindAvailableSpot(SpotType type) =>
        _spots.FirstOrDefault(s => s.Type == type && s.IsAvailable);

    public int AvailableCount(SpotType type) =>
        _spots.Count(s => s.Type == type && s.IsAvailable);
}

// ── Spot Type Resolver ────────────────────────────────
public static class SpotTypeResolver
{
    public static SpotType For(VehicleType v) => v switch
    {
        VehicleType.Motorcycle => SpotType.Compact,
        VehicleType.Car        => SpotType.Regular,
        VehicleType.Truck      => SpotType.Large,
        _ => throw new ArgumentOutOfRangeException()
    };
}

// ── Parking Lot (Singleton + Composite root) ──────────
public class ParkingLot
{
    private static readonly Lazy<ParkingLot> _instance =
        new(() => new ParkingLot());
    public static ParkingLot Instance => _instance.Value;

    private readonly List<ParkingFloor>          _floors  = new();
    private readonly Dictionary<Guid, Ticket>    _tickets = new();
    private readonly IPricingStrategy             _pricing;

    private ParkingLot(IPricingStrategy? pricing = null)
        => _pricing = pricing ?? new HourlyPricing();

    public void AddFloor(ParkingFloor floor) => _floors.Add(floor);

    public Ticket? CheckIn(Vehicle vehicle)
    {
        var spotType = SpotTypeResolver.For(vehicle.Type);
        var spot     = _floors
            .Select(f => f.FindAvailableSpot(spotType))
            .FirstOrDefault(s => s != null);

        if (spot == null)
        {
            Console.WriteLine($"No {spotType} spots available.");
            return null;
        }

        spot.Occupy();
        var ticket = new Ticket(vehicle, spot);
        _tickets[ticket.Id] = ticket;

        Console.WriteLine($"✅ Checked in {vehicle.LicensePlate} → Spot {spot.Id} (Ticket: {ticket.Id})");
        return ticket;
    }

    public decimal? CheckOut(Guid ticketId)
    {
        if (!_tickets.TryGetValue(ticketId, out var ticket))
        {
            Console.WriteLine("Ticket not found.");
            return null;
        }

        decimal fee = _pricing.Calculate(ticket);
        ticket.Settle(fee);
        ticket.Spot.Vacate();
        _tickets.Remove(ticketId);

        Console.WriteLine($"🚗 Checked out {ticket.Vehicle.LicensePlate}. Fee: ${fee:F2}");
        return fee;
    }

    public void PrintAvailability()
    {
        foreach (var floor in _floors)
        {
            Console.WriteLine($"Floor {floor.FloorNumber}:");
            foreach (SpotType t in Enum.GetValues<SpotType>())
                Console.WriteLine($"  {t}: {floor.AvailableCount(t)} available");
        }
    }
}
