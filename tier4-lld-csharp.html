Movie Ticket Booking
Optimistic Locking State (Seat) Strategy (Pricing) Command (Booking)
Key Design Decisions
  • Seat status: Available → TempHeld → Booked — timeout releases TempHeld
  • Concurrent seat selection uses lock + version check (optimistic concurrency)
  • Booking has a 10-minute payment window; a timer auto-cancels unpaid bookings
  • Pricing strategy: base rate × seat category multiplier + dynamic demand factor
  • Theatre → Screen → Show → Seat hierarchy keeps concerns separated
  • Payment result confirms or rolls back the booking atomically
Key Classes
  • Seat — row, number, category (Recliner/Premium/Regular), status
  • Show — movie, screen, date/time, seat map
  • Booking — user, seats, show, status, expiry timer
  • BookingService — holdSeats, confirmPayment, cancel
  • IPricingStrategy — flat, category-based, surge
  • Payment — gateway abstraction, idempotency key
Interviewer will ask: Q: How do you prevent two users booking the same seat simultaneously? → Seat hold is inside a lock(seat) block. Check status == Available, then atomically set to TempHeld with a timestamp. Only one thread wins; others see TempHeld and fail fast. In distributed systems, use a Redis SETNX lock with TTL. Q: What happens if the user doesn't pay within 10 minutes? → A Timer fires on booking creation. On expiry, status flips to Cancelled and each seat is reset to Available. In production, a background job scans for expired bookings — more reliable than in-process timers. Q: How do you handle partial seat selection (user wants 3 seats together)? → FindConsecutiveSeats() scans the seat map row-by-row for N adjacent Available seats. The entire group is held atomically inside a single lock to avoid partial holds.
Patterns: Optimistic locking — seat hold State — seat lifecycle Strategy — pricing Command — booking/cancel
📖 Concept Primer Optimistic vs Pessimistic Locking — concurrency without deadlocks
Pessimistic locking blocks everyone while one thread works. Optimistic locking lets everyone try, then validates at commit time — great when conflicts are rare (most ticket bookings).
// ── PESSIMISTIC: lock the resource upfront ─────────────
private readonly object _lock = new();
public bool TryHoldPessimistic(Seat seat)
{
    lock (_lock)                      // blocks all other threads
    {
        if (seat.Status != SeatStatus.Available) return false;
        seat.Status = SeatStatus.TempHeld;
        return true;
    }
}

// ── OPTIMISTIC: read freely, validate before commit ────
public bool TryHoldOptimistic(Seat seat, int expectedVersion)
{
    // No lock — read current state
    if (seat.Status != SeatStatus.Available) return false;

    // Atomic compare-and-swap on version
    int original = Interlocked.CompareExchange(
        ref seat._version, expectedVersion + 1, expectedVersion);

    if (original != expectedVersion)
        return false; // someone else changed it — retry

    seat.Status = SeatStatus.TempHeld;
    return true;
}

    public IEnumerable<Seat> GetAvailable() =>
        Seats.Where(s => s.Status == SeatStatus.Available);

    // Find N consecutive seats in the same row
    public List<Seat>? FindConsecutive(int count)
    {
        var byRow = Seats
            .Where(s => s.Status == SeatStatus.Available)
            .GroupBy(s => s.Row);

        foreach (var row in byRow)
        {
            var sorted = row.OrderBy(s => s.Number).ToList();
            for (int i = 0; i <= sorted.Count - count; i++)
            {
                bool consecutive = true;
                for (int j = 1; j < count; j++)
                    if (sorted[i+j].Number != sorted[i+j-1].Number + 1)
                        { consecutive = false; break; }
                if (consecutive) return sorted.GetRange(i, count);
            }
        }
        return null;
    }
}

// ── Booking Service (core orchestrator) ────────────────────────────────
public class BookingService
{
    private readonly IPricingStrategy            _pricing;
    private readonly IPaymentGateway             _payment;
    private readonly Dictionary<Guid, Booking>    _bookings = new();
    private readonly object                        _seatLock = new();

    public BookingService(IPricingStrategy pricing, IPaymentGateway payment)
        => (_pricing, _payment) = (pricing, payment);

    // Step 1: Hold seats temporarily (10-min window)
    public Booking? HoldSeats(User user, Show show, List<Seat> requestedSeats)
    {
        lock (_seatLock)
        {
            // Verify all seats still available
            if (requestedSeats.Any(s => s.Status != SeatStatus.Available))
            {
                Console.WriteLine("❌ One or more seats already taken.");
                return null;
            }

            decimal total = requestedSeats.Sum(s => _pricing.Price(show, s));

            // Mark seats as held
            foreach (var seat in requestedSeats)
            {
                seat.Status  = SeatStatus.TempHeld;
                seat.HeldAt  = DateTime.Now;
            }

            var booking = new Booking(user, show, requestedSeats, total, ExpireBooking);
            _bookings[booking.Id] = booking;
            Console.WriteLine($"🎟  Held {requestedSeats.Count} seat(s) for {user.Name} — ${total:F2}. Pay within 10 min.");
            return booking;
        }
    }

    // Step 2: Confirm with payment
    public bool ConfirmPayment(Guid bookingId)
    {
        if (!_bookings.TryGetValue(bookingId, out var b)) return false;
        if (b.Status != BookingStatus.PendingPayment) { Console.WriteLine("❌ Booking not payable."); return false; }

        if (!_payment.Charge(b.User, b.TotalCost, b.Id))
        {
            ExpireBooking(b);
            return false;
        }

        b.Status = BookingStatus.Confirmed;
        foreach (var s in b.Seats) s.Status = SeatStatus.Booked;
        Console.WriteLine($"✅ Booking {b.Id.ToString()[..8]} confirmed! Seats: {string.Join(", ", b.Seats)}");
        return true;
    }

    public void Cancel(Guid bookingId)
    {
        if (!_bookings.TryGetValue(bookingId, out var b)) return;
        if (b.Status == BookingStatus.Confirmed) _payment.Refund(b.User, b.TotalCost, b.Id);
        ExpireBooking(b);
        Console.WriteLine($"🗑  Booking {b.Id.ToString()[..8]} cancelled.");
    }

    private void ExpireBooking(Booking b)
    {
        lock (_seatLock)
        {
            if (b.Status is BookingStatus.Confirmed or BookingStatus.Cancelled) return;
            b.Status = BookingStatus.Expired;
            foreach (var s in b.Seats) s.Status = SeatStatus.Available;
            Console.WriteLine($"⌛ Booking {b.Id.ToString()[..8]} expired — seats released.");
        }
    }
}

Ride Sharing System
Observer (Driver Matching) Strategy (Pricing) State (Ride Lifecycle) Factory (Vehicle Types)
Key Design Decisions
  • Ride state machine: Requested → DriverAssigned → Started → Completed / Cancelled
  • Driver matching: geo-proximity + rating + vehicle type filter
  • Surge pricing strategy reads real-time demand/supply ratio
  • Driver location updates push to a spatial index (simulated here with list)
  • Trip fare = base + (distance × perKmRate) + (time × perMinRate) × surgeFactor
  • Rating system updates driver/rider reputation after trip completion
Key Classes
  • Rider / Driver — profile, rating, current location
  • Vehicle — type (Bike/Auto/Cab/XL), plate, capacity
  • Ride — state machine, pickup, dropoff, fare, OTP
  • RideService — requestRide, assignDriver, start, complete
  • IPricingStrategy — standard, surge, toll-inclusive
  • IDriverMatcher — nearest, highest-rated, by vehicle type
Interviewer will ask: Q: How does driver matching work at scale? → In production: a geospatial index (PostGIS, Redis Geo, or S2 cells) answers "drivers within R km" in O(log N). For LLD: keep a list of active drivers, compute Haversine distance, sort by score = distance + rating bonus + acceptance rate. Q: How do you implement surge pricing? → Track demand (active requests) and supply (available drivers) in a grid cell. SurgeFactor = demand/supply clamped to [1.0, 3.0]. Recalculated every 5 minutes. User sees estimated fare before confirming. Q: How do you handle ride cancellations and refunds? → Free cancellation within 2 minutes or before driver arrival. After that, a flat cancellation fee is charged. Driver gets a portion of the fee as compensation. All transitions fire events so notifications go out automatically.
Patterns: State — ride lifecycle Strategy — surge pricing Observer — ride events Factory — vehicle types

// ── Location (value object) ────────────────────────────────────────────
public record Location(double Lat, double Lng)
{
    // Haversine distance in km
    public double DistanceTo(Location other)
    {
        const double R = 6371;
        double dLat = (Lat  - other.Lat)  * Math.PI / 180;
        double dLng = (Lng  - other.Lng)  * Math.PI / 180;
        double a = Math.Sin(dLat/2) * Math.Sin(dLat/2) +
                   Math.Cos(Lat * Math.PI/180) * Math.Cos(other.Lat * Math.PI/180) *
                   Math.Sin(dLng/2) * Math.Sin(dLng/2);
        return R * 2 * Math.Atan2(Math.Sqrt(a), Math.Sqrt(1 - a));
    }
}

// ── Pricing Strategy ───────────────────────────────────────────────────
public interface IPricingStrategy
{
    decimal Estimate(Ride ride);
    decimal Final(Ride ride, double actualKm, double durationMins);
}

public class SurgePricingStrategy : IPricingStrategy
{
    private readonly decimal _surgeFactor;

    private static readonly Dictionary<VehicleType, (decimal Base, decimal PerKm, decimal PerMin)> _rates = new()
    {
        [VehicleType.Bike]    = (20m, 5m, 0.5m),
        [VehicleType.Auto]    = (30m, 8m, 1.0m),
        [VehicleType.Cab]     = (50m, 12m, 1.5m),
        [VehicleType.CabXL]   = (80m, 16m, 2.0m),
        [VehicleType.Premium] = (120m, 20m, 2.5m),
    };

    public SurgePricingStrategy(decimal surgeFactor = 1.0m) => _surgeFactor = surgeFactor;

    public decimal Estimate(Ride ride)
    {
        var r = _rates[ride.VehicleType];
        return (r.Base + (decimal)ride.DistanceKm * r.PerKm) * _surgeFactor;
    }

    public decimal Final(Ride ride, double actualKm, double durationMins)
    {
        var r = _rates[ride.VehicleType];
        return (r.Base + (decimal)actualKm * r.PerKm + (decimal)durationMins * r.PerMin) * _surgeFactor;
    }
}

public class NearestDriverMatcher : IDriverMatcher
{
    public Driver? Match(Ride ride, IEnumerable<Driver> available)
    {
        return available
            .Where(d => d.Status == DriverStatus.Available
                      && d.VehicleType == ride.VehicleType)
            .OrderBy(d => d.Location.DistanceTo(ride.Pickup))
            .ThenByDescending(d => d.Rating)
            .FirstOrDefault();
    }
}

// ── Ride Service ───────────────────────────────────────────────────────
public class RideService
{
    private readonly List<Driver>           _drivers  = new();
    private readonly List<Ride>             _rides    = new();
    private readonly IDriverMatcher         _matcher;
    private readonly IPricingStrategy        _pricing;
    private readonly List<IRideEventHandler> _handlers = new();

    public RideService(IDriverMatcher matcher, IPricingStrategy pricing)
        => (_matcher, _pricing) = (matcher, pricing);

    public void RegisterDriver(Driver d)         => _drivers.Add(d);
    public void Subscribe(IRideEventHandler h)   => _handlers.Add(h);

    public Ride? RequestRide(Rider rider, Location pickup, Location dropoff, VehicleType type)
    {
        var ride = new Ride(rider, pickup, dropoff, type);
        ride.EstimatedFare = _pricing.Estimate(ride);
        _rides.Add(ride);
        _handlers.ForEach(h => h.OnRideRequested(ride));

        var driver = _matcher.Match(ride, _drivers);
        if (driver == null)
        {
            Console.WriteLine("❌ No drivers available.");
            ride.Status = RideStatus.Cancelled;
            return null;
        }

        ride.Driver    = driver;
        ride.Status    = RideStatus.DriverAssigned;
        driver.Status  = DriverStatus.OnRide;
        Console.WriteLine($"🚗 Estimated fare: ₹{ride.EstimatedFare:F2} (Surge: {(_pricing is SurgePricingStrategy s ? "active" : "none")})");
        _handlers.ForEach(h => h.OnDriverAssigned(ride));
        return ride;
    }

    public void StartRide(Guid rideId, string otp)
    {
        var ride = _rides.First(r => r.Id == rideId);
        if (ride.Otp != otp) { Console.WriteLine("❌ Wrong OTP."); return; }
        ride.Status    = RideStatus.Started;
        ride.StartedAt = DateTime.Now;
        Console.WriteLine($"▶️  Ride started. OTP verified.");
    }

    public void CompleteRide(Guid rideId, double actualKm, double durationMins)
    {
        var ride = _rides.First(r => r.Id == rideId);
        ride.FinalFare    = _pricing.Final(ride, actualKm, durationMins);
        ride.Status       = RideStatus.Completed;
        ride.CompletedAt  = DateTime.Now;
        ride.Driver!.Status = DriverStatus.Available;
        _handlers.ForEach(h => h.OnRideCompleted(ride));
    }

    public void RateDriver(Guid rideId, double stars)
    {
        var ride = _rides.First(r => r.Id == rideId);
        if (ride.Driver == null) return;
        ride.Driver.Rating = (ride.Driver.Rating * 9 + stars) / 10; // rolling avg
        Console.WriteLine($"⭐ {ride.Driver.Name} new rating: {ride.Driver.Rating:F2}");
    }
}

Airline Management System
State (Flight) Strategy (Pricing) Observer (Notifications) Composite (Itinerary)
Key Design Decisions
  • Flight states: Scheduled → Boarding → InAir → Landed → Cancelled / Delayed
  • Seat inventory per cabin class (Economy, Business, First)
  • Dynamic pricing: base fare × load factor (occupancy %) × days-before-departure
  • Itinerary = composite of 1 or more flights (direct + connecting)
  • Crew assignment checks certifications and hours; avoids double-booking
  • All flight status changes notify subscribed passengers and crew
Key Classes
  • Flight — route, aircraft, state machine, seat map
  • Aircraft — tail number, cabin config, total capacity
  • Reservation — passenger, seats, itinerary, fare
  • AirlineSystem — searchFlights, book, checkin, cancel
  • IDynamicPricing — load-based, early-bird, last-minute
  • Crew / CrewAssignment — role, certifications, scheduling
Interviewer will ask: Q: How does dynamic pricing work? → Three factors multiply: (1) load factor — price rises as seats fill (60% occupancy = 1.2×, 90% = 1.8×). (2) Days before departure — <7 days = 1.5×, >60 days = 0.8× (early bird). (3) Cabin class base rate. All configurable. Q: How do you handle flight cancellations and cascading rebooking? → Flight cancellation fires an event. Each affected reservation is marked Disrupted. Rebooking service searches for next available flights on the same route. Automatically rebooks when a seat is found; otherwise adds to a waitlist and notifies the passenger. Q: How does check-in work and what prevents duplicate boarding passes? → Check-in sets a seat to CheckedIn status with the passenger's ID. Boarding pass has a unique barcode = HMAC(flightId + seatId + passengerId). Gate scanner validates the HMAC and checks the seat isn't already boarded.
Patterns: State — flight lifecycle Strategy — dynamic pricing Observer — status alerts Composite — itinerary

public interface IDynamicPricing
{

public class LoadBasedPricing : IDynamicPricing
{
    private static readonly Dictionary<CabinClass, decimal> _base = new()
    { [CabinClass.Economy] = 4000m, [CabinClass.Business] = 18000m, [CabinClass.First] = 45000m };

    public decimal Calculate(Flight flight, CabinClass cabin)
    {
        double load     = flight.Aircraft.LoadFactor();
        double daysToDep = (flight.Departure - DateTime.Now).TotalDays;

        decimal loadMul = load switch
        {
            >= 0.90 => 1.8m,
            >= 0.75 => 1.4m,
            >= 0.60 => 1.2m,
            _       => 1.0m
        };

        decimal daysMul = daysToDep switch
        {
            <= 3   => 1.8m,
            <= 7   => 1.5m,
            <= 14  => 1.2m,
            >= 60  => 0.8m,
            _      => 1.0m
        };

        return Math.Round(_base[cabin] * loadMul * daysMul, 0);
    }
}

public class StackOverflowService
{
    private readonly List<Question>     _questions = new();
    private readonly List<Tag>          _tags       = new();
    private readonly ReputationService  _repSvc     = new();

    public Question AskQuestion(SoUser user, string title, string body, List<string> tagNames)
    {
        if (user.Reputation < 1) throw new InvalidOperationException("Insufficient reputation.");
        var q = new Question(title, body, user);

        foreach (var name in tagNames.Take(5))
        {
            var tag = _tags.FirstOrDefault(t => t.Name == name) ?? new Tag(name);
            if (!_tags.Contains(tag)) _tags.Add(tag);
            tag.QuestionCount++;
            q.Tags.Add(tag);
        }

        _questions.Add(q);
        Console.WriteLine($"❓ [{string.Join(", ", tagNames)}] {user.Username} asked: \"{title}\"");
        return q;
    }

    public Answer PostAnswer(SoUser user, Question question, string body)
    {
        if (question.IsClosed) throw new InvalidOperationException("Question is closed.");
        var ans = new Answer(body, user);
        question.Answers.Add(ans);
        question.UpdatedAt = DateTime.Now;
        Console.WriteLine($"💬 {user.Username} answered question \"{question.Title}\"");
        return ans;
    }

    public void Vote(SoUser voter, Post post, VoteDirection direction)
    {
        if (!voter.CanVote) { Console.WriteLine($"❌ {voter.Username}: need 15 rep to vote."); return; }
        if (direction == VoteDirection.Down && !voter.CanDownvote) { Console.WriteLine($"❌ {voter.Username}: need 125 rep to downvote."); return; }
        if (post.Author.Id == voter.Id) { Console.WriteLine("❌ Cannot vote own post."); return; }

        // Idempotent: undo if same direction, switch if different
        if (post.Votes.TryGetValue(voter.Id, out var existing))
        {
            post.VoteScore += existing == VoteDirection.Up ? -1 : +1; // undo
            post.Votes.Remove(voter.Id);
            if (existing == direction) { Console.WriteLine($"↩️  Vote undone."); return; }
        }

        post.Votes[voter.Id] = direction;
        post.VoteScore      += direction == VoteDirection.Up ? 1 : -1;

        if (direction == VoteDirection.Up) _repSvc.OnUpvote(post.Author);
        else                               _repSvc.OnDownvote(post.Author, voter);
        Console.WriteLine($"👍 {voter.Username} {direction}voted. Score now: {post.VoteScore}");
    }

    public void AcceptAnswer(SoUser asker, Question q, Answer answer)
    {
        if (asker.Id != q.Author.Id) { Console.WriteLine("❌ Only the asker can accept."); return; }
        if (q.Accepted != null) q.Accepted.IsAccepted = false; // unaccept previous
        answer.IsAccepted = true;
        q.Accepted        = answer;
        _repSvc.OnAnswerAccepted(answer.Author, asker);
        Console.WriteLine($"✅ Answer by {answer.Author.Username} accepted for \"{q.Title}\"");
    }

    public IEnumerable<Question> Search(string query)
    {
        query = query.ToLower();
        return _questions
            .Where(q => !q.IsClosed)
            .Select(q => new {
                Q     = q,
                Score = q.SearchScore(
                    q.Title.ToLower().Contains(query) ||
                    q.Tags.Any(t => t.Name.ToLower().Contains(query)))
            })
            .OrderByDescending(x => x.Score)
            .Select(x => x.Q);
    }
}

// ── Order Book (per ticker) ─────────────────────────────────────────────
public class OrderBook
{
    public string Ticker { get; }

    // Bids: highest price first (max-heap via negated price)
    private readonly PriorityQueue<Order, (decimal, DateTime)> _bids =
        new(Comparer<(decimal, DateTime)>.Create((a, b) =>
            a.Item1 != b.Item1 ? b.Item1.CompareTo(a.Item1) : a.Item2.CompareTo(b.Item2)));

    // Asks: lowest price first (min-heap)
    private readonly PriorityQueue<Order, (decimal, DateTime)> _asks =
        new(Comparer<(decimal, DateTime)>.Create((a, b) =>
            a.Item1 != b.Item1 ? a.Item1.CompareTo(b.Item1) : a.Item2.CompareTo(b.Item2)));

    public decimal LastPrice { get; private set; }

    public OrderBook(string ticker) => Ticker = ticker;

    public void AddOrder(Order order)
    {
        decimal priority = order.Type == OrderType.Market
            ? (order.Side == OrderSide.Buy ? decimal.MaxValue : decimal.MinValue)
            : order.LimitPrice!.Value;

        if (order.Side == OrderSide.Buy)
            _bids.Enqueue(order, (priority, order.PlacedAt));
        else
            _asks.Enqueue(order, (priority, order.PlacedAt));
    }

    // Returns list of trades generated by matching
    public List<Trade> Match()
    {
        var trades = new List<Trade>();

        while (_bids.Count > 0 && _asks.Count > 0)
        {
            var bid = _bids.Peek();
            var ask = _asks.Peek();

            // Check if prices cross
            decimal bidPx = bid.LimitPrice ?? decimal.MaxValue;
            decimal askPx = ask.LimitPrice ?? decimal.MinValue;
            if (bidPx < askPx) break; // no match possible

            // Trade at the resting order's price
            decimal tradePrice  = bid.PlacedAt < ask.PlacedAt ? bidPx : askPx;
            int     tradeQty    = Math.Min(bid.RemainingQty, ask.RemainingQty);

            bid.FilledQty += tradeQty;
            ask.FilledQty += tradeQty;
            LastPrice      = tradePrice;

            trades.Add(new Trade(bid.Id, ask.Id, Ticker, tradeQty, tradePrice, DateTime.Now));
            Console.WriteLine($"🤝 TRADE {Ticker}: {tradeQty} @ ₹{tradePrice:F2}");

            if (bid.RemainingQty == 0) { bid.Status = OrderStatus.Filled;    _bids.Dequeue(); }
            else                          bid.Status = OrderStatus.PartiallyFilled;
            if (ask.RemainingQty == 0) { ask.Status = OrderStatus.Filled;    _asks.Dequeue(); }
            else                          ask.Status = OrderStatus.PartiallyFilled;
        }

        return trades;
    }
}

// ── Portfolio ──────────────────────────────────────────────────────────
public class Portfolio
{
    public Guid    AccountId  { get; }
    public string  Owner      { get; }
    public decimal Cash       { get; set; }
    public Dictionary<string, int>     Holdings   { get; } = new();
    public Dictionary<string, decimal> AvgCost    { get; } = new(); // for P&L

    public Portfolio(string owner, decimal initialCash)
        => (AccountId, Owner, Cash) = (Guid.NewGuid(), owner, initialCash);

    public void ApplyTrade(Trade trade, OrderSide side)
    {
        decimal value = trade.Price * trade.Quantity;
        if (side == OrderSide.Buy)
        {
            Cash -= value;
            Holdings[trade.Ticker]  = Holdings.GetValueOrDefault(trade.Ticker) + trade.Quantity;
            decimal prevQty    = Holdings[trade.Ticker] - trade.Quantity;
            decimal prevCost   = AvgCost.GetValueOrDefault(trade.Ticker);
            AvgCost[trade.Ticker] = (prevQty * prevCost + value) / Holdings[trade.Ticker];
        }
        else
        {
            Cash += value;
            Holdings[trade.Ticker] -= trade.Quantity;
        }
        Console.WriteLine($"💼 {Owner}: {side} {trade.Quantity} {trade.Ticker} @ ₹{trade.Price:F2}. Cash: ₹{Cash:F2}");
    }

    public void PrintSummary(Dictionary<string, decimal> marketPrices)
    {
        Console.WriteLine($"\n📊 Portfolio: {Owner} | Cash: ₹{Cash:F2}");
        foreach (var (ticker, qty) in Holdings.Where(h => h.Value > 0))
        {
            decimal mkt   = marketPrices.GetValueOrDefault(ticker, AvgCost[ticker]);
            decimal pnl   = (mkt - AvgCost[ticker]) * qty;
            Console.WriteLine($"  {ticker,-8} {qty,5} shares  avg ₹{AvgCost[ticker]:F2}  mkt ₹{mkt:F2}  P&L: {(pnl >= 0 ? "+" : "")}₹{pnl:F2}");
        }
    }
}

// ── Brokerage Service (Facade) ─────────────────────────────────────────
public class BrokerageService
{
    private readonly Dictionary<string, OrderBook>  _books     = new();
    private readonly Dictionary<Guid, Portfolio>    _portfolios= new();
    private readonly Dictionary<Guid, Order>        _orders    = new();
    private readonly MarketDataFeed                 _feed;

    public BrokerageService(MarketDataFeed feed) => _feed = feed;

    public void OpenAccount(Portfolio p) => _portfolios[p.AccountId] = p;

    private OrderBook GetBook(string ticker)
    {
        if (!_books.ContainsKey(ticker)) _books[ticker] = new OrderBook(ticker);
        return _books[ticker];
    }

    public Order? PlaceOrder(Portfolio portfolio, OrderSide side, OrderType type,
                             string ticker, int qty, decimal? limit = null)
    {
        // Pre-trade checks
        if (side == OrderSide.Buy)
        {
            decimal requiredCash = (limit ?? _feed.GetPrice(ticker)) * qty;
            if (portfolio.Cash < requiredCash) { Console.WriteLine($"❌ Insufficient cash."); return null; }
        }
        else
        {
            int holdings = portfolio.Holdings.GetValueOrDefault(ticker);
            if (holdings < qty) { Console.WriteLine($"❌ Insufficient shares ({holdings} held)."); return null; }
        }

        var order = new Order(portfolio.AccountId, ticker, side, type, qty, limit);
        _orders[order.Id] = order;
        Console.WriteLine($"📋 {portfolio.Owner}: {side} {qty} {ticker} {(type == OrderType.Market ? "MARKET" : $"LIMIT ₹{limit}")}");

        var book   = GetBook(ticker);
        book.AddOrder(order);

        // Try to match immediately
        var trades = book.Match();
        foreach (var trade in trades)
        {
            // Apply to buyer's and seller's portfolios
            var buyOrder  = _orders[trade.BuyOrderId];
            var sellOrder = _orders[trade.SellOrderId];
            if (_portfolios.TryGetValue(buyOrder.AccountId,  out var buyer))  buyer.ApplyTrade(trade, OrderSide.Buy);
            if (_portfolios.TryGetValue(sellOrder.AccountId, out var seller)) seller.ApplyTrade(trade, OrderSide.Sell);
            _feed.PublishTick(ticker, trade.Price);
        }

        return order;
    }
}

// ── Innings ────────────────────────────────────────────────────────────
public class Innings
{
    public Team                                  BattingTeam  { get; }
    public Team                                  BowlingTeam  { get; }
    public List<Ball>                            Balls        { get; } = new();
    public Dictionary<Guid, BattingCard>        BattingCards { get; } = new();
    public Dictionary<Guid, BowlingCard>        BowlingCards { get; } = new();
    public bool                                  IsComplete   { get; set; }
    public int?                                  Target       { get; set; }

    public int    TotalRuns    => Balls.Sum(b => b.TotalRuns);
    public int    Wickets      => Balls.Count(b => b.IsWicket);
    public int    LegalBalls   => Balls.Count(b => b.IsLegal);
    public double Overs        => LegalBalls / 6.0;
    public double RunRate      => Overs == 0 ? 0 : TotalRuns / Overs;
    public string Score        => $"{TotalRuns}/{Wickets} ({LegalBalls / 6}.{LegalBalls % 6} ov)";

    public Innings(Team batting, Team bowling) => (BattingTeam, BowlingTeam) = (batting, bowling);

    public void AddBall(Ball ball)
    {
        Balls.Add(ball);

        // Update batting stats
        if (!BattingCards.ContainsKey(ball.Batsman.Id))
            BattingCards[ball.Batsman.Id] = new(ball.Batsman);
        var bc = BattingCards[ball.Batsman.Id];
        bc.Runs       += ball.RunsScored;
        bc.BallsFaced += ball.IsLegal ? 1 : 0;
        bc.Fours      += ball.RunsScored == 4 ? 1 : 0;
        bc.Sixes      += ball.RunsScored == 6 ? 1 : 0;
        if (ball.IsWicket) bc.HowOut = ball.Dismissal;

        // Update bowling stats
        if (!BowlingCards.ContainsKey(ball.Bowler.Id))
            BowlingCards[ball.Bowler.Id] = new(ball.Bowler);
        var bwl = BowlingCards[ball.Bowler.Id];
        bwl.LegalBalls += ball.IsLegal ? 1 : 0;
        bwl.RunsGiven  += ball.TotalRuns;
        bwl.Wickets    += ball.IsWicket ? 1 : 0;
        bwl.Wides      += ball.Extra == ExtraType.Wide ? 1 : 0;
        bwl.NoBalls    += ball.Extra == ExtraType.NoBall ? 1 : 0;
    }
}

// ── DLS Strategy ───────────────────────────────────────────────────────
public interface IDlsStrategy
{
    int RecalculateTarget(int originalTarget, double oversRemaining, int wicketsLost);
}

public class SimplifiedDls : IDlsStrategy
{
    // Resource percentage table (simplified — real DLS uses full tables)
    private static readonly double[] _resourcePct = { 0, 7.6, 17.2, 28.6, 40.9, 52.4, 62.7, 71.8, 79.9, 87.2, 93.8, 100 };

    public int RecalculateTarget(int originalTarget, double oversRemaining, int wicketsLost)
    {
        int    idx              = Math.Min((int)oversRemaining, 11);
        double resourcesRemaining = _resourcePct[idx] * (1.0 - wicketsLost * 0.09);
        int    adjustedTarget   = (int)(originalTarget * resourcesRemaining / 100) + 1;
        Console.WriteLine($"🌧️  DLS: {oversRemaining:F1} overs left, {wicketsLost} wkts down. New target: {adjustedTarget}");
        return adjustedTarget;
    }
}

public class CricketMatch
{
    public Guid         Id      { get; } = Guid.NewGuid();
    public MatchFormat  Format  { get; }
    public Team         HomeTeam{ get; }
    public Team         AwayTeam{ get; }
    public DateTime     Date    { get; }
    public string       Venue   { get; }
    public List<Innings> Innings { get; } = new();
    public MatchPhase   Phase   { get; private set; } = MatchPhase.PreMatch;
    public string       Result  { get; private set; } = "";
    public int          MaxOvers{ get; }

    private readonly List<IScoreboardObserver> _observers = new();
    private readonly IDlsStrategy             _dls;

    public CricketMatch(MatchFormat format, Team home, Team away, DateTime date, string venue, IDlsStrategy? dls = null)
    {
        (Format, HomeTeam, AwayTeam, Date, Venue) = (format, home, away, date, venue);
        MaxOvers = format == MatchFormat.T20 ? 20 : format == MatchFormat.ODI ? 50 : 0;
        _dls = dls ?? new SimplifiedDls();
    }

    public void Subscribe(IScoreboardObserver obs) => _observers.Add(obs);

    public Innings StartInnings(Team batting, Team bowling)
    {
        var phase = Innings.Count == 0 ? MatchPhase.Innings1 : MatchPhase.Innings2;
        ChangePhase(phase);
        var innings = new Innings(batting, bowling);
        if (Innings.Count > 0) innings.Target = Innings[0].TotalRuns + 1;
        Innings.Add(innings);
        return innings;
    }

    public void DeliverBall(Innings innings, Ball ball)
    {
        innings.AddBall(ball);
        foreach (var obs in _observers) obs.OnBall(innings, ball);
    }

    public void ApplyRainInterruption(Innings innings, double oversLost)
    {
        double remaining = MaxOvers - innings.Overs - oversLost;
        if (innings.Target.HasValue)
            innings.Target = _dls.RecalculateTarget(innings.Target.Value, remaining, innings.Wickets);
    }

    public void DeclareResult(string result)
    {
        Result = result;
        ChangePhase(MatchPhase.Result);
        Console.WriteLine($"\n🏆 RESULT: {result}");
    }

    public void PrintScorecard()
    {
        foreach (var inn in Innings)
        {
            Console.WriteLine($"\n📋 {inn.BattingTeam.Name} — {inn.Score}");
            Console.WriteLine("  Batting:");
            foreach (var bc in inn.BattingCards.Values)
                Console.WriteLine($"  {bc.Player.Name,-20} {bc.Runs,4} ({bc.BallsFaced}b) {bc.Fours}×4 {bc.Sixes}×6 SR:{bc.StrikeRate:F1} [{bc.HowOut}]");
            Console.WriteLine("  Bowling:");
            foreach (var bw in inn.BowlingCards.Values)
                Console.WriteLine($"  {bw.Player.Name,-20} {bw.OversStr,5} ov  {bw.Wickets}/{bw.RunsGiven}  Econ:{bw.Economy:F2}");
        }
    }

    private void ChangePhase(MatchPhase p)
    {
        Phase = p;
        foreach (var obs in _observers) obs.OnPhaseChanged(p);
    }
}
